5   package net.sourceforge.pmd.document;
6   
7   /**
8    * Represents a file which contains programming code that will be fixed.
9    */
10  public interface Document {
11  
12      /**
13       * Insert a text at a line at the position/column specified. If there is any text to the right of the insertion,
14       * that text is shifted by the length of the text to insert, which means that it is not replaced.
15       * @param beginLine the line in which to insert the text
16       * @param beginColumn the position in the line in which to insert the text
17       * @param textToInsert the text to be added
18       */
19      void insert(int beginLine, int beginColumn, String textToInsert);
20  
21      /**
22       * Replace a specific region in the document which contains text by another text, which not necessarily is the same
23       * length as the region's one.
24       * @param regionByOffset the region in which a text will be inserted to replace the current document's contents
25       * @param textToReplace the text to insert
26       */
27      void replace(RegionByLine regionByOffset, String textToReplace);
28  
29      /**
30       * Delete a region in the document, removing all text which contains it. If there is any text to the right of this
31       * region, it will be shifted to the left by the length of the region to delete.
32       * @param regionByOffset the region in which to erase all the text
33       */
34      void delete(RegionByLine regionByOffset);
35  }
