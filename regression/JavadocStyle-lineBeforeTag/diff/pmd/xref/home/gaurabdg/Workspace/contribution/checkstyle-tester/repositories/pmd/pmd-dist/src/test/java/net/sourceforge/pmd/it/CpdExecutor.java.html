5   package net.sourceforge.pmd.it;
6   
7   import java.nio.charset.StandardCharsets;
8   import java.nio.file.Path;
9   import java.util.Arrays;
10  
11  import org.apache.commons.io.IOUtils;
12  import org.apache.commons.lang3.SystemUtils;
13  
14  import net.sourceforge.pmd.PMDVersion;
15  
16  /**
17   * Executes CPD from command line. Deals with the differences, when CPD is run on Windows or on Linux.
18   *
19   * @author Andreas Dangel
20   */
21  public class CpdExecutor {
22      private static final String PMD_BIN_PREFIX = "pmd-bin-";
23  
24      private CpdExecutor() {
25          // this is a helper class only
26      }
27  
28      private static ExecutionResult runCpdUnix(Path tempDir, String ... arguments) throws Exception {
29          String cmd = tempDir.resolve(PMD_BIN_PREFIX + PMDVersion.VERSION + "/bin/run.sh").toAbsolutePath().toString();
30          ProcessBuilder pb = new ProcessBuilder(cmd, "cpd");
31          pb.command().addAll(Arrays.asList(arguments));
32          pb.redirectErrorStream(true);
33          Process process = pb.start();
34          String output = IOUtils.toString(process.getInputStream(), StandardCharsets.UTF_8);
35  
36          int result = process.waitFor();
37          return new ExecutionResult(result, output, null, null);
38      }
39  
40      private static ExecutionResult runCpdWindows(Path tempDir, String ... arguments) throws Exception {
41          String cmd = tempDir.resolve(PMD_BIN_PREFIX + PMDVersion.VERSION + "/bin/cpd.bat").toAbsolutePath().toString();
42          ProcessBuilder pb = new ProcessBuilder(cmd);
43          pb.command().addAll(Arrays.asList(arguments));
44          pb.redirectErrorStream(true);
45          Process process = pb.start();
46          String output = IOUtils.toString(process.getInputStream(), StandardCharsets.UTF_8);
47  
48          int result = process.waitFor();
49          return new ExecutionResult(result, output, null, null);
50      }
51  
52      /**
53       * Executes CPD found in tempDir with the given command line arguments.
54       * @param tempDir the directory, to which the binary distribution has been extracted
55       * @param arguments the arguments to execute CPD with
56       * @return collected result of the execution
57       * @throws Exception if the execution fails for any reason (executable not found, ...)
58       */
59      public static ExecutionResult runCpd(Path tempDir, String ... arguments) throws Exception {
60          if (SystemUtils.IS_OS_WINDOWS) {
61              return runCpdWindows(tempDir, arguments);
62          } else {
63              return runCpdUnix(tempDir, arguments);
64          }
65      }
66  }
