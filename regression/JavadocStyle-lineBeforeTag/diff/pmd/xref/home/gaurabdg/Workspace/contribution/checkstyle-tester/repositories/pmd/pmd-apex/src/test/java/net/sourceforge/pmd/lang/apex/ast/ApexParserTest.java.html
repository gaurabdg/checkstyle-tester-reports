113     @Test
114     public void checkComments() {
115 
116         String code = "public  /** Comment on Class */ class SimpleClass {\n" // line 1
117             + "    /** Comment on m1 */"
118             + "    public void method1() {\n" // line 2
119             + "    }\n" // line 3
120             + "    public void method2() {\n" // line 4
121             + "    }\n" // line 5
122             + "}\n"; // line 6
123 
124         ApexNode<Compilation> root = parse(code);
125 
126         assertThat(root, instanceOf(ASTUserClass.class));
127         ApexNode<?> comment = root.getChild(0);
128         assertThat(comment, instanceOf(ASTFormalComment.class));
129 
130         assertNotEquals(comment.getNode(), null);
131         assertThat(comment.getNode(), instanceOf(ASTFormalComment.AstComment.class));
132         assertPosition(comment, 1, 9, 1, 31);
133         assertEquals("/** Comment on Class */", ((ASTFormalComment) comment).getToken());
134 
135         ApexNode<?> m1 = root.getChild(2);
136         assertThat(m1, instanceOf(ASTMethod.class));
137 
138         ApexNode<?> comment2 = m1.getChild(0);
139         assertThat(comment2, instanceOf(ASTFormalComment.class));
140         assertEquals("/** Comment on m1 */", ((ASTFormalComment) comment2).getToken());
141     }

174     @Test
175     public void stackOverflowDuringClassParsing() throws Exception {
176         String source = IOUtils.toString(ApexParserTest.class.getResourceAsStream("StackOverflowClass.cls"),
177                 StandardCharsets.UTF_8);
178         ApexNode<Compilation> rootNode = parse(source);
179         Assert.assertNotNull(rootNode);
180 
181         int count = visitPosition(rootNode, 0);
182         Assert.assertEquals(427, count);
183     }
184 
185     private int visitPosition(Node node, int count) {
186         int result = count + 1;
187         Assert.assertTrue(node.getBeginLine() > 0);
188         Assert.assertTrue(node.getBeginColumn() > 0);
189         Assert.assertTrue(node.getEndLine() > 0);
190         Assert.assertTrue(node.getEndColumn() > 0);
191         for (int i = 0; i < node.getNumChildren(); i++) {
192             result = visitPosition(node.getChild(i), result);
193         }
194         return result;
195     }

197     // TEST HELPER

199     private static void assertPosition(Node node, int beginLine, int beginColumn, int endLine, int endColumn) {
200         assertEquals("Wrong begin line", beginLine, node.getBeginLine());
201         assertEquals("Wrong begin column", beginColumn, node.getBeginColumn());
202         assertEquals("Wrong end line", endLine, node.getEndLine());
203         assertEquals("Wrong end column", endColumn, node.getEndColumn());
204     }
