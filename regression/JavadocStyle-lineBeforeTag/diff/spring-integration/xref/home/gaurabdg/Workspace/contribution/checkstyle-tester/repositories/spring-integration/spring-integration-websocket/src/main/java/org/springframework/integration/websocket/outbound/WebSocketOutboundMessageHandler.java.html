17  package org.springframework.integration.websocket.outbound;
18  
19  import java.util.ArrayList;
20  import java.util.List;
21  import java.util.ListIterator;
22  
23  import org.springframework.integration.handler.AbstractMessageHandler;
24  import org.springframework.integration.support.json.JacksonPresent;
25  import org.springframework.integration.websocket.ClientWebSocketContainer;
26  import org.springframework.integration.websocket.IntegrationWebSocketContainer;
27  import org.springframework.integration.websocket.support.PassThruSubProtocolHandler;
28  import org.springframework.integration.websocket.support.SubProtocolHandlerRegistry;
29  import org.springframework.messaging.Message;
30  import org.springframework.messaging.MessageHandlingException;
31  import org.springframework.messaging.converter.ByteArrayMessageConverter;
32  import org.springframework.messaging.converter.CompositeMessageConverter;
33  import org.springframework.messaging.converter.DefaultContentTypeResolver;
34  import org.springframework.messaging.converter.MappingJackson2MessageConverter;
35  import org.springframework.messaging.converter.MessageConverter;
36  import org.springframework.messaging.converter.StringMessageConverter;
37  import org.springframework.messaging.simp.SimpMessageHeaderAccessor;
38  import org.springframework.messaging.simp.SimpMessageType;
39  import org.springframework.util.Assert;
40  import org.springframework.util.CollectionUtils;
41  import org.springframework.util.MimeTypeUtils;
42  import org.springframework.web.socket.WebSocketSession;
43  import org.springframework.web.socket.handler.SessionLimitExceededException;
44  
45  /**
46   * @author Artem Bilan
47   *
48   * @since 4.1
49   */
50  public class WebSocketOutboundMessageHandler extends AbstractMessageHandler {
51  
52  	private final List<MessageConverter> defaultConverters = new ArrayList<>(3);
53  
54  	{
55  		this.defaultConverters.add(new StringMessageConverter());
56  		this.defaultConverters.add(new ByteArrayMessageConverter());
57  		if (JacksonPresent.isJackson2Present()) {
58  			DefaultContentTypeResolver resolver = new DefaultContentTypeResolver();
59  			resolver.setDefaultMimeType(MimeTypeUtils.APPLICATION_JSON);
60  			MappingJackson2MessageConverter converter = new MappingJackson2MessageConverter();
61  			converter.setContentTypeResolver(resolver);
62  			this.defaultConverters.add(converter);
63  		}
64  	}
65  
66  	private final CompositeMessageConverter messageConverter = new CompositeMessageConverter(this.defaultConverters);
67  
68  	private final IntegrationWebSocketContainer webSocketContainer;
69  
70  	private final SubProtocolHandlerRegistry subProtocolHandlerRegistry;
71  
72  	private final boolean client;
73  
74  	private List<MessageConverter> messageConverters;
75  
76  	private boolean mergeWithDefaultConverters = false;
77  
78  	public WebSocketOutboundMessageHandler(IntegrationWebSocketContainer webSocketContainer) {
79  		this(webSocketContainer, new SubProtocolHandlerRegistry(new PassThruSubProtocolHandler()));
80  	}
81  
82  	public WebSocketOutboundMessageHandler(IntegrationWebSocketContainer webSocketContainer,
83  			SubProtocolHandlerRegistry protocolHandlerRegistry) {
84  
85  		Assert.notNull(webSocketContainer, "'webSocketContainer' must not be null");
86  		Assert.notNull(protocolHandlerRegistry, "'protocolHandlerRegistry' must not be null");
87  		this.webSocketContainer = webSocketContainer;
88  		this.client = webSocketContainer instanceof ClientWebSocketContainer;
89  		this.subProtocolHandlerRegistry = protocolHandlerRegistry;
90  		List<String> subProtocols = protocolHandlerRegistry.getSubProtocols();
91  		this.webSocketContainer.addSupportedProtocols(subProtocols.toArray(new String[0]));
92  	}
93  
94  	/**
95  	 * Set the message converters to use. These converters are used to convert the message to send for appropriate
96  	 * internal subProtocols type.
97  	 * @param messageConverters The message converters.
98  	 */
99  	public void setMessageConverters(List<MessageConverter> messageConverters) {
100 		Assert.noNullElements(messageConverters.toArray(), "'messageConverters' must not contain null entries");
101 		this.messageConverters = new ArrayList<>(messageConverters);
102 	}
103 
104 
105 	/**
106 	 * Flag which determines if the default converters should be available after
107 	 * custom converters.
108 	 * @param mergeWithDefaultConverters true to merge, false to replace.
109 	 */
110 	public void setMergeWithDefaultConverters(boolean mergeWithDefaultConverters) {
111 		this.mergeWithDefaultConverters = mergeWithDefaultConverters;
112 	}
113 
114 	@Override
115 	public String getComponentType() {
116 		return "websocket:outbound-channel-adapter";
117 	}
118 
119 	@Override
120 	protected void onInit() {
121 		super.onInit();
122 		if (!CollectionUtils.isEmpty(this.messageConverters)) {
123 			List<MessageConverter> converters = this.messageConverter.getConverters();
124 			if (this.mergeWithDefaultConverters) {
125 				ListIterator<MessageConverter> iterator =
126 						this.messageConverters.listIterator(this.messageConverters.size());
127 				while (iterator.hasPrevious()) {
128 					MessageConverter converter = iterator.previous();
129 					converters.add(0, converter);
130 				}
131 			}
132 			else {
133 				converters.clear();
134 				converters.addAll(this.messageConverters);
135 			}
136 		}
137 	}
138 
139 	@Override
140 	protected void handleMessageInternal(Message<?> message) {
141 		String sessionId = null;
142 		if (!this.client) {
143 			sessionId = this.subProtocolHandlerRegistry.resolveSessionId(message);
144 			if (sessionId == null) {
145 				throw new IllegalArgumentException("The WebSocket 'sessionId' is required in the MessageHeaders");
146 			}
147 		}
148 		WebSocketSession session = this.webSocketContainer.getSession(sessionId);
149 		try {
150 			SimpMessageHeaderAccessor headers = SimpMessageHeaderAccessor.wrap(message);
151 			headers.setLeaveMutable(true);
152 			headers.setMessageTypeIfNotSet(SimpMessageType.MESSAGE);
153 			Object payload = message.getPayload();
154 			Message<?> messageToSend =
155 					this.messageConverter.toMessage(payload, headers.getMessageHeaders());
156 			Assert.state(messageToSend != null,
157 					() -> "The message converter '" + this.messageConverter +
158 							"' produced no message to send based on the request message: '" + message + "'");
159 			this.subProtocolHandlerRegistry.findProtocolHandler(session).handleMessageToClient(session, messageToSend);
160 		}
161 		catch (SessionLimitExceededException ex) {
162 			try {
163 				logger.error("Terminating session id '" + sessionId + "'", ex);
164 				this.webSocketContainer.closeSession(session, ex.getStatus());
165 			}
166 			catch (Exception secondException) {
167 				logger.error("Exception terminating session id '" + sessionId + "'", secondException);
168 			}
169 		}
170 		catch (Exception e) {
171 			throw new MessageHandlingException(message, "Failed to handle message in the [" + this + ']', e);
172 		}
173 	}
174 
175 }
