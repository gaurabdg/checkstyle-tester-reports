50  /**
51   * Base class for outbound Web Service-invoking Messaging Gateways.
52   *
53   * @author Mark Fisher
54   * @author Jonas Partner

58   */
59  public abstract class AbstractWebServiceOutboundGateway extends AbstractReplyProducingMessageHandler {
60  
61  	protected final DefaultUriBuilderFactory uriFactory = new DefaultUriBuilderFactory(); // NOSONAR - final
62  
63  	private final String uri;
64  
65  	private final DestinationProvider destinationProvider;
66  
67  	private final Map<String, Expression> uriVariableExpressions = new HashMap<>();
68  
69  	private StandardEvaluationContext evaluationContext;
70  
71  	private WebServiceMessageCallback requestCallback;

73  	private WebServiceTemplate webServiceTemplate;
74  
75  	private boolean ignoreEmptyResponses = true;
76  
77  	private SoapHeaderMapper headerMapper = new DefaultSoapHeaderMapper();
78  
79  	private boolean webServiceTemplateExplicitlySet;
80  
81  	public AbstractWebServiceOutboundGateway(final String uri, WebServiceMessageFactory messageFactory) {
82  		Assert.hasText(uri, "URI must not be empty");
83  		this.webServiceTemplate = new WebServiceTemplate(messageFactory);
84  		this.destinationProvider = null;
85  		this.uri = uri;
86  	}
87  
88  	public AbstractWebServiceOutboundGateway(DestinationProvider destinationProvider,
89  			WebServiceMessageFactory messageFactory) {
90  
91  		Assert.notNull(destinationProvider, "DestinationProvider must not be null");
92  		this.webServiceTemplate = new WebServiceTemplate(messageFactory);
93  		this.destinationProvider = destinationProvider;
94  		// we always call WebServiceTemplate methods with an explicit URI argument,
95  		// but in case the WebServiceTemplate is accessed directly we'll set this:
96  		this.webServiceTemplate.setDestinationProvider(destinationProvider);
97  		this.uri = null;
98  	}

100 	public void setHeaderMapper(SoapHeaderMapper headerMapper) {
101 		Assert.notNull(headerMapper, "'headerMapper' must not be null");
102 		this.headerMapper = headerMapper;
103 	}

105 	/**
106 	 * Set the Map of URI variable expressions to evaluate against the outbound message
107 	 * when replacing the variable placeholders in a URI template.
108 	 * @param uriVariableExpressions The URI variable expressions.
109 	 */
110 	public void setUriVariableExpressions(Map<String, Expression> uriVariableExpressions) {
111 		synchronized (this.uriVariableExpressions) {
112 			this.uriVariableExpressions.clear();
113 			this.uriVariableExpressions.putAll(uriVariableExpressions);
114 		}

117 	/**
118 	 * Specify whether the URI should be encoded after any <code>uriVariables</code>

121 	 * @see org.springframework.web.util.UriComponentsBuilder

127 		setEncodingMode(
128 				encodeUri
129 						? DefaultUriBuilderFactory.EncodingMode.TEMPLATE_AND_VALUES

133 	/**
134 	 * Set the encoding mode to use.
135 	 * By default this is set to {@link DefaultUriBuilderFactory.EncodingMode#TEMPLATE_AND_VALUES}.

137 	 * @since 5.3
138 	 */
139 	public void setEncodingMode(DefaultUriBuilderFactory.EncodingMode encodingMode) {
140 		Assert.notNull(encodingMode, "'encodingMode' must not be null");

142 	}

156 	}

163 		Assert.notNull(template, "'webServiceTemplate' must not be null");

165 		this.webServiceTemplateExplicitlySet = true;

172 	}

204 	}

206 	@Override
207 	protected void doInit() {
208 		this.evaluationContext = ExpressionUtils.createStandardEvaluationContext(this.getBeanFactory());
209 		Assert.state(this.destinationProvider == null || CollectionUtils.isEmpty(this.uriVariableExpressions),
210 				"uri variables are not supported when a DestinationProvider is supplied.");
211 	}
212 
213 	protected WebServiceTemplate getWebServiceTemplate() {
214 		return this.webServiceTemplate;
215 	}
216 
217 	@Override
218 	public final Object handleRequestMessage(Message<?> requestMessage) {
219 		URI uriWithVariables = prepareUri(requestMessage);
220 		if (uriWithVariables == null) {
221 			throw new MessageDeliveryException(requestMessage, "Failed to determine URI for " +
222 					"Web Service request in outbound gateway: " + this.getComponentName());
223 		}
224 		Object responsePayload = doHandle(uriWithVariables.toString(), requestMessage, this.requestCallback);
225 		if (responsePayload != null && !(this.ignoreEmptyResponses
226 				&& responsePayload instanceof String
227 				&& !StringUtils.hasText((String) responsePayload))) {
228 
229 			return responsePayload;
230 		}
231 		return null;
232 	}
233 
234 	private URI prepareUri(Message<?> requestMessage) {
235 		if (this.destinationProvider != null) {
236 			return this.destinationProvider.getDestination();
237 		}
238 
239 		Map<String, Object> uriVariables =
240 				ExpressionEvalMap.from(this.uriVariableExpressions)
241 						.usingEvaluationContext(this.evaluationContext)
242 						.withRoot(requestMessage)
243 						.build();
244 
245 		return this.uriFactory.expand(this.uri, uriVariables);
246 	}
247 
248 
249 	protected abstract Object doHandle(String theUri, Message<?> requestMessage,
250 			WebServiceMessageCallback reqCallback);
251 
252 	protected abstract class RequestMessageCallback extends TransformerObjectSupport
253 			implements WebServiceMessageCallback {
254 
255 		private final WebServiceMessageCallback reqCallback;
256 
257 		private final Message<?> requestMessage;
258 
259 		public RequestMessageCallback(WebServiceMessageCallback requestCallback, Message<?> requestMessage) {
260 			this.reqCallback = requestCallback;
261 			this.requestMessage = requestMessage;
262 		}
263 
264 		@Override
265 		public void doWithMessage(WebServiceMessage message) throws IOException, TransformerException {
266 			Object payload = this.requestMessage.getPayload();
267 			doWithMessageInternal(message, payload);
268 			if (message instanceof SoapMessage) {
269 				AbstractWebServiceOutboundGateway.this.headerMapper
270 						.fromHeadersToRequest(this.requestMessage.getHeaders(), (SoapMessage) message);
271 			}
272 			if (this.reqCallback != null) {
273 				this.reqCallback.doWithMessage(message);
274 			}
275 		}
276 
277 		public abstract void doWithMessageInternal(WebServiceMessage message, Object payload)
278 				throws IOException, TransformerException;
279 
280 	}
281 
282 	protected abstract class ResponseMessageExtractor extends TransformerObjectSupport
283 			implements WebServiceMessageExtractor<Object> {
284 
285 		@Override
286 		public Object extractData(WebServiceMessage message)
287 				throws IOException, TransformerException {
288 
289 			Object resultObject = this.doExtractData(message);
290 
291 			if (resultObject != null && message instanceof SoapMessage) {
292 				Map<String, Object> mappedMessageHeaders =
293 						AbstractWebServiceOutboundGateway.this.headerMapper.toHeadersFromReply((SoapMessage) message);
294 				return getMessageBuilderFactory()
295 						.withPayload(resultObject)
296 						.copyHeaders(mappedMessageHeaders)
297 						.build();
298 			}
299 			else {
300 				return resultObject;
301 			}
302 		}
303 
304 		public abstract Object doExtractData(WebServiceMessage message) throws IOException, TransformerException;
305 
306 	}
