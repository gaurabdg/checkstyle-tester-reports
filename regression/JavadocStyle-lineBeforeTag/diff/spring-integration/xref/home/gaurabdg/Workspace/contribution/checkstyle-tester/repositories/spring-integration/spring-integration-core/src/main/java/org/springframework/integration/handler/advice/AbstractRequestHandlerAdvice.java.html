17  package org.springframework.integration.handler.advice;
18  
19  import java.lang.reflect.Method;
20  
21  import org.aopalliance.intercept.MethodInterceptor;
22  import org.aopalliance.intercept.MethodInvocation;
23  
24  import org.springframework.aop.ProxyMethodInvocation;
25  import org.springframework.integration.context.IntegrationObjectSupport;
26  import org.springframework.messaging.Message;
27  import org.springframework.messaging.MessageHandler;
28  import org.springframework.messaging.MessagingException;
29  
30  /**
31   * Base class for {@link MessageHandler} advice classes. Subclasses should provide an
32   * implementation for {@link #doInvoke(ExecutionCallback, Object, Message)}. Used to
33   * advise the handleRequestMessage method for
34   * {@link org.springframework.integration.handler.AbstractReplyProducingMessageHandler} or
35   * {@link MessageHandler#handleMessage(Message)} for other message handlers.
36   *
37   * @author Gary Russell
38   * @author Artem Bilan
39   * @since 2.2
40   */
41  public abstract class AbstractRequestHandlerAdvice extends IntegrationObjectSupport
42  		implements MethodInterceptor {
43  
44  	@Override
45  	public final Object invoke(final MethodInvocation invocation) throws Throwable {
46  
47  		Method method = invocation.getMethod();
48  		Object[] arguments = invocation.getArguments();
49  		boolean isMessageMethod = (method.getName().equals("handleRequestMessage") || method.getName().equals("handleMessage"))
50  				&& (arguments.length == 1 && arguments[0] instanceof Message);
51  
52  		Object invocationThis = invocation.getThis();
53  		if (!isMessageMethod) {
54  			boolean isMessageHandler = invocationThis != null
55  					&& MessageHandler.class.isAssignableFrom(invocationThis.getClass());
56  			if (!isMessageHandler && this.logger.isWarnEnabled()) {
57  				String clazzName = invocationThis == null ? method.getDeclaringClass().getName() : invocationThis.getClass().getName();
58  				this.logger.warn("This advice " + this.getClass().getName() +
59  						" can only be used for MessageHandlers; an attempt to advise method '" + method.getName() +
60  						"' in '" + clazzName + "' is ignored");
61  			}
62  			return invocation.proceed();
63  		}
64  		else {
65  			Message<?> message = (Message<?>) arguments[0];
66  			try {
67  				return doInvoke(new CallbackImpl(invocation), invocationThis, message);
68  			}
69  			catch (Exception e) {
70  				throw this.unwrapThrowableIfNecessary(e);
71  			}
72  		}
73  	}
74  
75  	/**
76  	 * Subclasses implement this method to apply behavior to the {@link MessageHandler}.
77  	 * <p>
78  	 * callback.execute() invokes the handler method and returns its result, or null.
79  	 *
80  	 * @param callback Subclasses invoke the execute() method on this interface to invoke the handler method.
81  	 * @param target   The target handler.
82  	 * @param message  The message that will be sent to the handler.
83  	 * @return the result after invoking the {@link MessageHandler}.
84  	 */
85  	protected abstract Object doInvoke(ExecutionCallback callback, Object target, Message<?> message);
86  
87  	/**
88  	 * Unwrap the cause of a {@link AbstractRequestHandlerAdvice.ThrowableHolderException}.
89  	 * @param e The exception.
90  	 * @return The cause, or e, if not a {@link AbstractRequestHandlerAdvice.ThrowableHolderException}
91  	 */
92  	protected Exception unwrapExceptionIfNecessary(Exception e) {
93  		Exception actualException = e;
94  		if (e instanceof ThrowableHolderException
95  				&& e.getCause() instanceof Exception) {
96  			actualException = (Exception) e.getCause();
97  		}
98  		return actualException;
99  	}
100 
101 	/**
102 	 * Unwrap the cause of a {@link AbstractRequestHandlerAdvice.ThrowableHolderException}.
103 	 * @param e The exception.
104 	 * @return The cause, or e, if not a {@link AbstractRequestHandlerAdvice.ThrowableHolderException}
105 	 */
106 	protected Throwable unwrapThrowableIfNecessary(Exception e) {
107 		Throwable actualThrowable = e;
108 		if (e instanceof ThrowableHolderException) {
109 			actualThrowable = e.getCause();
110 		}
111 		return actualThrowable;
112 	}
113 
114 	/**
115 	 * Called by subclasses in doInvoke() to proceed() the invocation. Callers
116 	 * unwrap {@link AbstractRequestHandlerAdvice.ThrowableHolderException}s and use
117 	 * the cause for evaluation and re-throwing purposes.
118 	 * See {@link AbstractRequestHandlerAdvice#unwrapExceptionIfNecessary(Exception)}.
119 	 */
120 	protected interface ExecutionCallback {
121 
122 		/**
123 		 * Call this for a normal invocation.proceed().
124 		 *
125 		 * @return The result of the execution.
126 		 */
127 		Object execute();
128 
129 		/**
130 		 * Call this when it is necessary to clone the invocation before
131 		 * calling proceed() - such as when the invocation might be called
132 		 * multiple times - for example in a retry advice.
133 		 *
134 		 * @return The result of the execution.
135 		 */
136 		Object cloneAndExecute();
137 
138 	}
139 
140 	private static final class CallbackImpl implements ExecutionCallback {
141 
142 		private final MethodInvocation invocation;
143 
144 		CallbackImpl(MethodInvocation invocation) {
145 			this.invocation = invocation;
146 		}
147 
148 		@Override
149 		public Object execute() {
150 			try {
151 				return this.invocation.proceed();
152 			}
153 			catch (Throwable e) { //NOSONAR - ok to catch; unwrapped and rethrown below
154 				throw new ThrowableHolderException(e);
155 			}
156 		}
157 
158 		@Override
159 		public Object cloneAndExecute() {
160 			try {
161 				/*
162 	 			* If we don't copy the invocation carefully it won't keep a reference to the other
163 	 			* interceptors in the chain.
164 	 			*/
165 				if (this.invocation instanceof ProxyMethodInvocation) {
166 					return ((ProxyMethodInvocation) this.invocation).invocableClone().proceed();
167 				}
168 				else {
169 					throw new IllegalStateException(
170 							"MethodInvocation of the wrong type detected - this should not happen with Spring AOP," +
171 									" so please raise an issue if you see this exception");
172 				}
173 			}
174 			catch (Exception e) { //NOSONAR - catch necessary so we can wrap Errors
175 				throw new MessagingException((Message<?>) this.invocation.getArguments()[0], "Failed to handle", e);
176 			}
177 			catch (Throwable e) { //NOSONAR - ok to catch; unwrapped and rethrown below
178 				throw new ThrowableHolderException(e);
179 			}
180 		}
181 
182 	}
183 
184 	@SuppressWarnings("serial")
185 	protected static final class ThrowableHolderException extends RuntimeException {
186 
187 		ThrowableHolderException(Throwable cause) {
188 			super(cause);
189 		}
190 
191 	}
192
