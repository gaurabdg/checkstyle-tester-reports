17  package org.springframework.integration.mongodb.inbound;
18  
19  import java.util.List;
20  
21  import org.springframework.data.mongodb.MongoDatabaseFactory;
22  import org.springframework.data.mongodb.core.MongoOperations;
23  import org.springframework.data.mongodb.core.MongoTemplate;
24  import org.springframework.data.mongodb.core.convert.MongoConverter;
25  import org.springframework.data.mongodb.core.query.BasicQuery;
26  import org.springframework.data.mongodb.core.query.Query;
27  import org.springframework.expression.Expression;
28  import org.springframework.expression.TypeLocator;
29  import org.springframework.expression.common.LiteralExpression;
30  import org.springframework.expression.spel.support.StandardEvaluationContext;
31  import org.springframework.expression.spel.support.StandardTypeLocator;
32  import org.springframework.integration.endpoint.AbstractMessageSource;
33  import org.springframework.integration.expression.ExpressionUtils;
34  import org.springframework.integration.mongodb.support.MongoHeaders;
35  import org.springframework.integration.support.AbstractIntegrationMessageBuilder;
36  import org.springframework.integration.transaction.IntegrationResourceHolder;
37  import org.springframework.lang.Nullable;
38  import org.springframework.transaction.support.TransactionSynchronizationManager;
39  import org.springframework.util.Assert;
40  import org.springframework.util.CollectionUtils;
41  
42  import com.mongodb.DBObject;
43  
44  /**
45   * An instance of {@link org.springframework.integration.core.MessageSource} which returns
46   * a {@link org.springframework.messaging.Message} with a payload which is the result of
47   * execution of a {@link Query}. When expectSingleResult is false (default), the MongoDb
48   * {@link Query} is executed using {@link MongoOperations#find(Query, Class)} method which
49   * returns a {@link List}. The returned {@link List} will be used as the payload of the
50   * {@link org.springframework.messaging.Message} returned by the {{@link #receive()}
51   * method. An empty {@link List} is treated as null, thus resulting in no
52   * {@link org.springframework.messaging.Message} returned by the {{@link #receive()}
53   * method.
54   * <p>
55   * When expectSingleResult is true, the {@link MongoOperations#findOne(Query, Class)} is
56   * used instead, and the message payload will be the single object returned from the
57   * query.
58   *
59   * @author Amol Nayak
60   * @author Oleg Zhurakousky
61   * @author Yaron Yamin
62   * @author Artem Bilan
63   * @author Gary Russell
64   *
65   * @since 2.2
66   */
67  public class MongoDbMessageSource extends AbstractMessageSource<Object> {
68  
69  	@Nullable
70  	private final MongoDatabaseFactory mongoDbFactory;
71  
72  	private final Expression queryExpression;
73  
74  	private Expression collectionNameExpression = new LiteralExpression("data");
75  
76  	private StandardEvaluationContext evaluationContext;
77  
78  	private MongoOperations mongoTemplate;
79  
80  	private MongoConverter mongoConverter;
81  
82  	private Class<?> entityClass = DBObject.class;
83  
84  	private boolean expectSingleResult = false;
85  
86  	private volatile boolean initialized = false;
87  
88  	/**
89  	 * Creates an instance with the provided {@link MongoDatabaseFactory} and SpEL expression
90  	 * which should resolve to a MongoDb 'query' string (see https://www.mongodb.org/display/DOCS/Querying).
91  	 * The 'queryExpression' will be evaluated on every call to the {@link #receive()} method.
92  	 * @param mongoDbFactory The mongodb factory.
93  	 * @param queryExpression The query expression.
94  	 */
95  	public MongoDbMessageSource(MongoDatabaseFactory mongoDbFactory, Expression queryExpression) {
96  		Assert.notNull(mongoDbFactory, "'mongoDbFactory' must not be null");
97  		Assert.notNull(queryExpression, "'queryExpression' must not be null");
98  		this.mongoDbFactory = mongoDbFactory;
99  		this.queryExpression = queryExpression;
100 	}
101 
102 	/**
103 	 * Creates an instance with the provided {@link MongoOperations} and SpEL expression
104 	 * which should resolve to a Mongo 'query' string (see https://www.mongodb.org/display/DOCS/Querying).
105 	 * It assumes that the {@link MongoOperations} is fully initialized and ready to be used.
106 	 * The 'queryExpression' will be evaluated on every call to the {@link #receive()} method.
107 	 * @param mongoTemplate The mongo template.
108 	 * @param queryExpression The query expression.
109 	 */
110 	public MongoDbMessageSource(MongoOperations mongoTemplate, Expression queryExpression) {
111 		Assert.notNull(mongoTemplate, "'mongoTemplate' must not be null");
112 		Assert.notNull(queryExpression, "'queryExpression' must not be null");
113 		this.mongoDbFactory = null;
114 		this.mongoTemplate = mongoTemplate;
115 		this.queryExpression = queryExpression;
116 	}
117 
118 	/**
119 	 * Allows you to set the type of the entityClass that will be passed to the
120 	 * {@link MongoTemplate#find(Query, Class)} or {@link MongoTemplate#findOne(Query, Class)} method.
121 	 * Default is {@link DBObject}.
122 	 * @param entityClass The entity class.
123 	 */
124 	public void setEntityClass(Class<?> entityClass) {
125 		Assert.notNull(entityClass, "'entityClass' must not be null");
126 		this.entityClass = entityClass;
127 	}
128 
129 	/**
130 	 * Allows you to manage which find* method to invoke on {@link MongoTemplate}.
131 	 * Default is 'false', which means the {@link #receive()} method will use
132 	 * the {@link MongoTemplate#find(Query, Class)} method. If set to 'true',
133 	 * {@link #receive()} will use {@link MongoTemplate#findOne(Query, Class)},
134 	 * and the payload of the returned {@link org.springframework.messaging.Message}
135 	 * will be the returned target Object of type
136 	 * identified by {@link #entityClass} instead of a List.
137 	 * @param expectSingleResult true if a single result is expected.
138 	 */
139 	public void setExpectSingleResult(boolean expectSingleResult) {
140 		this.expectSingleResult = expectSingleResult;
141 	}
142 
143 	/**
144 	 * Sets the SpEL {@link Expression} that should resolve to a collection name
145 	 * used by the {@link Query}. The resulting collection name will be included
146 	 * in the {@link MongoHeaders#COLLECTION_NAME} header.
147 	 * @param collectionNameExpression The collection name expression.
148 	 */
149 	public void setCollectionNameExpression(Expression collectionNameExpression) {
150 		Assert.notNull(collectionNameExpression, "'collectionNameExpression' must not be null");
151 		this.collectionNameExpression = collectionNameExpression;
152 	}
153 
154 	/**
155 	 * Allows you to provide a custom {@link MongoConverter} used to assist in deserialization
156 	 * data read from MongoDb. Only allowed if this instance was constructed with a
157 	 * {@link MongoDatabaseFactory}.
158 	 * @param mongoConverter The mongo converter.
159 	 */
160 	public void setMongoConverter(MongoConverter mongoConverter) {
161 		Assert.isNull(this.mongoTemplate,
162 				"'mongoConverter' can not be set when instance was constructed with MongoTemplate");
163 		this.mongoConverter = mongoConverter;
164 	}
165 
166 	@Override
167 	public String getComponentType() {
168 		return "mongo:inbound-channel-adapter";
169 	}
170 
171 	@Override
172 	protected void onInit() {
173 		this.evaluationContext =
174 				ExpressionUtils.createStandardEvaluationContext(getBeanFactory());
175 		TypeLocator typeLocator = this.evaluationContext.getTypeLocator();
176 		if (typeLocator instanceof StandardTypeLocator) {
177 			//Register MongoDB query API package so FQCN can be avoided in query-expression.
178 			((StandardTypeLocator) typeLocator).registerImport("org.springframework.data.mongodb.core.query");
179 		}
180 		if (this.mongoDbFactory != null) {
181 			this.mongoTemplate = new MongoTemplate(this.mongoDbFactory, this.mongoConverter);
182 		}
183 		this.initialized = true;
184 	}
185 
186 	/**
187 	 * Will execute a {@link Query} returning its results as the Message payload.
188 	 * The payload can be either {@link List} of elements of objects of type
189 	 * identified by {@link #entityClass}, or a single element of type identified by {@link #entityClass}
190 	 * based on the value of {@link #expectSingleResult} attribute which defaults to 'false' resulting
191 	 * {@link org.springframework.messaging.Message} with payload of type
192 	 * {@link List}. The collection name used in the
193 	 * query will be provided in the {@link MongoHeaders#COLLECTION_NAME} header.
194 	 */
195 	@Override
196 	protected Object doReceive() {
197 		Assert.isTrue(this.initialized, "This class is not yet initialized. Invoke its afterPropertiesSet() method");
198 		AbstractIntegrationMessageBuilder<Object> messageBuilder = null;
199 		Object value = this.queryExpression.getValue(this.evaluationContext);
200 		Assert.notNull(value, "'queryExpression' must not evaluate to null");
201 		Query query;
202 		if (value instanceof String) {
203 			query = new BasicQuery((String) value);
204 		}
205 		else if (value instanceof Query) {
206 			query = ((Query) value);
207 		}
208 		else {
209 			throw new IllegalStateException("'queryExpression' must evaluate to String " +
210 					"or org.springframework.data.mongodb.core.query.Query");
211 		}
212 
213 		Assert.notNull(query, "'queryExpression' must not evaluate to null");
214 		String collectionName = this.collectionNameExpression.getValue(this.evaluationContext, String.class);
215 		Assert.notNull(collectionName, "'collectionNameExpression' must not evaluate to null");
216 
217 		Object result = null;
218 		if (this.expectSingleResult) {
219 			result = this.mongoTemplate.findOne(query, this.entityClass, collectionName);
220 		}
221 		else {
222 			List<?> results = this.mongoTemplate.find(query, this.entityClass, collectionName);
223 			if (!CollectionUtils.isEmpty(results)) {
224 				result = results;
225 			}
226 		}
227 		if (result != null) {
228 			messageBuilder = this.getMessageBuilderFactory().withPayload(result)
229 					.setHeader(MongoHeaders.COLLECTION_NAME, collectionName);
230 		}
231 
232 		Object holder = TransactionSynchronizationManager.getResource(this);
233 		if (holder != null) {
234 			Assert.isInstanceOf(IntegrationResourceHolder.class, holder);
235 			((IntegrationResourceHolder) holder).addAttribute("mongoTemplate", this.mongoTemplate);
236 		}
237 
238 		return messageBuilder;
239 	}
