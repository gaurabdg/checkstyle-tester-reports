52  /**
53   * Implementation of {@link org.springframework.messaging.MessageHandler} which writes
54   * Message data into a Redis store identified by a key {@link String}.
55   *
56   * It supports the collection types identified by {@link CollectionType}.
57   *
58   * It supports batch updates or single item entry.
59   *
60   * "Batch updates" means that the payload of the Message may be a Map or Collection. With
61   * such a payload, individual items from it are added to the corresponding Redis store.
62   * See {@link #handleMessageInternal(Message)} for more details.
63   *
64   * You can instead choose to persist such a payload as a single item if the
65   * {@link #extractPayloadElements} property is set to false (default is true).
66   *
67   * @author Oleg Zhurakousky
68   * @author Gary Russell
69   * @author Mark Fisher
70   * @author Artem Bilan
71   *
72   * @since 2.2
73   */
74  public class RedisStoreWritingMessageHandler extends AbstractMessageHandler {

242 	@Override
243 	protected void onInit() {
244 		this.evaluationContext =
245 				ExpressionUtils.createStandardEvaluationContext(this.getBeanFactory());
246 		Assert.state(!this.mapKeyExpressionExplicitlySet ||
247 						(this.collectionType == CollectionType.MAP || this.collectionType == CollectionType.PROPERTIES),
248 				"'mapKeyExpression' can only be set for CollectionType.MAP or CollectionType.PROPERTIES");
249 		if (!this.redisTemplateExplicitlySet) {
250 			if (!this.extractPayloadElements) {
251 				RedisTemplate<String, Object> template = new RedisTemplate<>();
252 				StringRedisSerializer serializer = new StringRedisSerializer();
253 				template.setKeySerializer(serializer);
254 				template.setHashKeySerializer(serializer);
255 				this.redisTemplate = template;
256 			}
257 			this.redisTemplate.setConnectionFactory(this.connectionFactory);
258 			this.redisTemplate.afterPropertiesSet();
259 		}
260 		this.initialized = true;
261 	}

263 	/**
264 	 * Will extract the payload from the Message and store it in the collection identified by the
265 	 * key (which may be determined by an expression). The type of collection is specified by the
266 	 * {@link #collectionType} property. The default CollectionType is LIST.
267 	 * <p>
268 	 * The rules for storing the payload are:
269 	 * <p>
270 	 * <b>LIST/SET</b>
271 	 * If the payload is of type Collection and {@link #extractPayloadElements} is 'true' (default),
272 	 * the payload will be added using the addAll() method. If {@link #extractPayloadElements}
273 	 * is set to 'false', then regardless of the payload type, the payload will be added using add().
274 	 * <p>
275 	 * <b>ZSET</b>
276 	 * In addition to the rules described for LIST/SET, ZSET allows 'score' information
277 	 * to be provided. The score can be provided using the {@link RedisHeaders#ZSET_SCORE} message header
278 	 * when the payload is not a Map, or by sending a Map as the payload where each Map 'key' is a
279 	 * value to be saved and each corresponding Map 'value' is the score assigned to it.
280 	 * If {@link #extractPayloadElements} is set to 'false' the map will be stored as a single entry.
281 	 * If the 'score' can not be determined, the default value (1) will be used.
282 	 * <p>
283 	 * <b>MAP/PROPERTIES</b>
284 	 * You can also add items to a Map or Properties based store.
285 	 * If the payload itself is of type Map or Properties, it can be stored either as a batch or single
286 	 * item following the same rules as described above for other collection types.
287 	 * If the payload itself needs to be stored as a value of the map/property then the map key
288 	 * must be specified via the mapKeyExpression (default {@link RedisHeaders#MAP_KEY} Message header).
289 	 */
290 	@SuppressWarnings("unchecked")
291 	@Override
292 	protected void handleMessageInternal(Message<?> message) {
293 		String key = this.keyExpression.getValue(this.evaluationContext, message, String.class);
294 		Assert.hasText(key, () -> "Failed to determine a key for the Redis store based on the message: " + message);
295 
296 		RedisStore store = createStoreView(key);
297 
298 		Assert.state(this.initialized,
299 				"handler not initialized - afterPropertiesSet() must be called before the first use");
300 		try {
301 			if (this.collectionType == CollectionType.ZSET) {
302 				writeToZset((RedisZSet<Object>) store, message);
303 			}
304 			else if (this.collectionType == CollectionType.SET) {
305 				writeToSet((RedisSet<Object>) store, message);
306 			}
307 			else if (this.collectionType == CollectionType.LIST) {
308 				writeToList((RedisList<Object>) store, message);
309 			}
310 			else if (this.collectionType == CollectionType.MAP) {
311 				writeToMap((RedisMap<Object, Object>) store, message);
312 			}
313 			else if (this.collectionType == CollectionType.PROPERTIES) {
314 				writeToProperties((RedisProperties) store, message);
315 			}
316 		}
317 		catch (Exception ex) {
318 			throw IntegrationUtils.wrapInHandlingExceptionIfNecessary(message,
319 					() -> "Failed to store Message data into Redis collection in the [" + this + ']', ex);
320 		}
321 	}

323 	@SuppressWarnings("unchecked")
324 	private void writeToZset(RedisZSet<Object> zset, final Message<?> message) {
325 		final Object payload = message.getPayload();
326 		final BoundZSetOperations<String, Object> ops =
327 				(BoundZSetOperations<String, Object>) this.redisTemplate.boundZSetOps(zset.getKey());
328 		boolean zsetIncrementHeader = extractZsetIncrementHeader(message);
329 		if (this.extractPayloadElements) {
330 			if ((payload instanceof Map<?, ?> && this.verifyAllMapValuesOfTypeNumber((Map<?, ?>) payload))) {
331 				Map<Object, Number> payloadAsMap = (Map<Object, Number>) payload;
332 				processInPipeline(() -> {
333 					for (Entry<Object, Number> entry : payloadAsMap.entrySet()) {
334 						Number d = entry.getValue();
335 						incrementOrOverwrite(ops, entry.getKey(), d == null ?
336 										determineScore(message) :
337 										NumberUtils.convertNumberToTargetClass(d, Double.class),
338 								zsetIncrementHeader);
339 					}
340 				});
341 			}
342 			else if (payload instanceof Collection<?>) {
343 				processInPipeline(() -> {
344 					for (Object object : ((Collection<?>) payload)) {
345 						incrementOrOverwrite(ops, object, determineScore(message), zsetIncrementHeader);
346 					}
347 				});
348 			}
349 			else {
350 				incrementOrOverwrite(ops, payload, this.determineScore(message), zsetIncrementHeader);
351 			}
352 		}
353 		else {
354 			incrementOrOverwrite(ops, payload, this.determineScore(message), zsetIncrementHeader);
355 		}
356 	}
357 
358 	private boolean extractZsetIncrementHeader(Message<?> message) {
359 		Boolean value = this.zsetIncrementScoreExpression.getValue(this.evaluationContext, message, Boolean.class);
360 		return value != null ? value : false;
361 	}

363 	private void writeToList(RedisList<Object> list, Message<?> message) {
364 		Object payload = message.getPayload();
365 		if (this.extractPayloadElements) {
366 			if (payload instanceof Collection<?>) {
367 				list.addAll((Collection<?>) payload);
368 			}
369 			else {
370 				list.add(payload);
371 			}
372 		}
373 		else {
374 			list.add(payload);
375 		}
376 	}

378 	@SuppressWarnings("unchecked")
379 	private void writeToSet(final RedisSet<Object> set, Message<?> message) {
380 		final Object payload = message.getPayload();
381 		if (this.extractPayloadElements && payload instanceof Collection<?>) {
382 			BoundSetOperations<String, Object> ops =
383 					(BoundSetOperations<String, Object>) this.redisTemplate.boundSetOps(set.getKey());
384 
385 			processInPipeline(() -> {
386 				for (Object object : ((Collection<?>) payload)) {
387 					ops.add(object);
388 				}
389 			});
390 		}
391 		else {
392 			set.add(payload);
393 		}
394 	}
395 
396 	private void writeToMap(final RedisMap<Object, Object> map, Message<?> message) {
397 		final Object payload = message.getPayload();
398 		if (this.extractPayloadElements && payload instanceof Map<?, ?>) {
399 			processInPipeline(() -> map.putAll((Map<?, ?>) payload));
400 		}
401 		else {
402 			Object key = this.determineMapKey(message, false);
403 			map.put(key, payload);
404 		}
405 	}
406 
407 	private void writeToProperties(final RedisProperties properties, Message<?> message) {
408 		final Object payload = message.getPayload();
409 		if (this.extractPayloadElements && payload instanceof Properties) {
410 			processInPipeline(() -> properties.putAll((Properties) payload));
411 		}
412 		else {
413 			Assert.isInstanceOf(String.class, payload, "For property, payload must be a String.");
414 			Object key = this.determineMapKey(message, true);
415 			properties.put(key, payload);
416 		}
417 	}
418 
419 	private void processInPipeline(PipelineCallback callback) {
420 		RedisConnectionFactory connectionFactoryForPipeline = this.redisTemplate.getConnectionFactory();
421 		Assert.state(connectionFactoryForPipeline != null, "RedisTemplate returned no connection factory");
422 		RedisConnection connection =
423 				RedisConnectionUtils.bindConnection(connectionFactoryForPipeline);
424 		try {
425 			connection.openPipeline();
426 			callback.process();
427 		}
428 		finally {
429 			connection.closePipeline();
430 			RedisConnectionUtils.unbindConnection(connectionFactoryForPipeline);
431 		}
432 	}

434 	private Object determineMapKey(Message<?> message, boolean property) {
435 		Object mapKey = this.mapKeyExpression.getValue(this.evaluationContext, message);
436 		Assert.notNull(mapKey, () -> "Cannot determine a map key for the entry based on the message: " + message);
437 		if (property) {
438 			Assert.isInstanceOf(String.class, mapKey, "For property, key must be a String");
439 		}
440 		return mapKey;
441 	}
442 
443 	private void incrementOrOverwrite(BoundZSetOperations<String, Object> ops, Object object, Double score,
444 			boolean zsetIncrementScore) {
445 		if (score != null) {
446 			doIncrementOrOverwrite(ops, object, score, zsetIncrementScore);
447 		}
448 		else {
449 			this.logger.debug("Zset Score could not be determined. Using default score of 1");
450 			doIncrementOrOverwrite(ops, object, 1d, zsetIncrementScore);
451 		}
452 	}
453 
454 	private void doIncrementOrOverwrite(BoundZSetOperations<String, Object> ops, Object object, Double score,
455 			boolean increment) {
456 		if (increment) {
457 			ops.incrementScore(object, score);
458 		}
459 		else {
460 			ops.add(object, score);
461 		}
462 	}
463 
464 	private boolean verifyAllMapValuesOfTypeNumber(Map<?, ?> map) {
465 		for (Object value : map.values()) {
466 			if (!(value instanceof Number)) {
467 				if (this.logger.isWarnEnabled()) {
468 					this.logger.warn("failed to extract payload elements because '" +
469 							value + "' is not of type Number");
470 				}
471 				return false;
472 			}
473 		}
474 		return true;
475 	}

477 	private RedisStore createStoreView(String key) {
478 		RedisCollectionFactoryBean fb = new RedisCollectionFactoryBean();
479 		fb.setKey(key);
480 		fb.setTemplate(this.redisTemplate);
481 		fb.setType(this.collectionType);
482 		fb.afterPropertiesSet();
483 		return fb.getObject();
484 	}
485 
486 	private double determineScore(Message<?> message) {
487 		Object scoreHeader = message.getHeaders().get(RedisHeaders.ZSET_SCORE);
488 		if (scoreHeader == null) {
489 			return 1d;
490 		}
491 		else {
492 			Assert.isInstanceOf(Number.class, scoreHeader,
493 					() -> "Header " + RedisHeaders.ZSET_SCORE + " must be a Number");
494 			Number score = (Number) scoreHeader;
495 			return Double.valueOf(score.toString());
496 		}
497 	}
