17  package org.springframework.integration.jpa.dsl;
18  
19  import org.springframework.integration.jpa.core.JpaExecutor;
20  import org.springframework.integration.jpa.support.PersistMode;
21  
22  /**
23   * A {@link JpaBaseOutboundEndpointSpec} extension for the {@code updating}
24   * {@link org.springframework.integration.jpa.outbound.JpaOutboundGateway} mode.
25   * The {@code outbound-channel-adapter} is achievable through an internal {@code producesReply} option.
26   *
27   * @author Artem Bilan
28   *
29   * @since 5.0
30   */
31  public class JpaUpdatingOutboundEndpointSpec extends JpaBaseOutboundEndpointSpec<JpaUpdatingOutboundEndpointSpec> {
32  
33  	protected JpaUpdatingOutboundEndpointSpec(JpaExecutor jpaExecutor) {
34  		super(jpaExecutor);
35  	}
36  
37  	protected JpaUpdatingOutboundEndpointSpec producesReply(boolean producesReply) {
38  		this.target.setProducesReply(producesReply);
39  		if (producesReply) {
40  			this.target.setRequiresReply(true);
41  		}
42  		return this;
43  	}
44  
45  	/**
46  	 * Specify a {@link PersistMode} for the gateway.
47  	 * Defaults to {@link PersistMode#MERGE}.
48  	 * @param persistMode the {@link PersistMode} to use.
49  	 * @return the spec
50  	 */
51  	public JpaUpdatingOutboundEndpointSpec persistMode(PersistMode persistMode) {
52  		this.jpaExecutor.setPersistMode(persistMode);
53  		return this;
54  	}
55  
56  	/**
57  	 * If set to {@code true} the {@link javax.persistence.EntityManager#flush()} will be called
58  	 * after persistence operation.
59  	 * Has the same effect, if the {@link #flushSize} is specified to {@code 1}.
60  	 * For convenience in cases when the provided entity to persist is not an instance of {@link Iterable}.
61  	 * @param flush defaults to {@code false}.
62  	 * @return the spec
63  	 */
64  	public JpaUpdatingOutboundEndpointSpec flush(boolean flush) {
65  		this.jpaExecutor.setFlush(flush);
66  		return this;
67  	}
68  
69  	/**
70  	 * If the provided value is greater than {@code 0}, then {@link javax.persistence.EntityManager#flush()}
71  	 * will be called after persistence operations as well as within batch operations.
72  	 * This property has precedence over the {@link #flush}, if it is specified to a value greater than {@code 0}.
73  	 * If the entity to persist is not an instance of {@link Iterable} and this property is greater than {@code 0},
74  	 * then the entity will be flushed as if the {@link #flush} attribute was set to {@code true}.
75  	 * @param flushSize defaults to {@code 0}.
76  	 * @return the spec
77  	 */
78  	public JpaUpdatingOutboundEndpointSpec flushSize(int flushSize) {
79  		this.jpaExecutor.setFlushSize(flushSize);
80  		return this;
81  	}
82  
83  	/**
84  	 * If set to {@code true} the {@link javax.persistence.EntityManager#clear()} will be called,
85  	 * and only if the {@link javax.persistence.EntityManager#flush()}
86  	 * was called after performing persistence operations.
87  	 * @param clearOnFlush defaults to {@code false}.
88  	 * @return the spec
89  	 */
90  	public JpaUpdatingOutboundEndpointSpec clearOnFlush(boolean clearOnFlush) {
91  		this.jpaExecutor.setClearOnFlush(clearOnFlush);
92  		return this;
93  	}
94  
95  }
