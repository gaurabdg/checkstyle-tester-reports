17  package org.springframework.integration.context;
18  
19  /**
20   * Interface for components that wish to be considered for
21   * an orderly shutdown using management interfaces. beforeShutdown()
22   * will be called before schedulers, executors etc, are stopped.
23   * afterShutdown() is called after the shutdown delay.
24   *
25   * @author Gary Russell
26   * @since 2.2
27   *
28   */
29  public interface OrderlyShutdownCapable {
30  
31  	/**
32  	 * Called before shutdown begins. Implementations should
33  	 * stop accepting new messages. Can optionally return the
34  	 * number of active messages in process.
35  	 * @return The number of active messages if available.
36  	 */
37  	int beforeShutdown();
38  
39  	/**
40  	 * Called after normal shutdown of schedulers, executors etc,
41  	 * and after the shutdown delay has elapsed, but before any
42  	 * forced shutdown of any remaining active scheduler/executor
43  	 * threads.Can optionally return the number of active messages
44  	 * still in process.
45  	 * @return The number of active messages if available.
46  	 */
47  	int afterShutdown();
48  
49  }
