54  /**
55   * A message handler that executes an SQL update. Dynamic query parameters are supported through the
56   * {@link SqlParameterSourceFactory} abstraction, the default implementation of which wraps the message so that its bean
57   * properties can be referred to by name in the query string, e.g.
58   *
59   * <pre class="code">
60   * INSERT INTO FOOS (MESSAGE_ID, PAYLOAD) VALUES (:headers[id], :payload)
61   * </pre>
62   *
63   * <p>
64   * When a message payload is an instance of {@link Iterable}, a
65   * {@link NamedParameterJdbcOperations#batchUpdate(String, SqlParameterSource[])} is performed, where each
66   * {@link SqlParameterSource} instance is based on items wrapped into an internal {@link Message} implementation with
67   * headers from the request message.
68   * <p>
69   * When a {@link #preparedStatementSetter} is configured, it is applied for each item in the appropriate
70   * {@link JdbcOperations#batchUpdate(String, BatchPreparedStatementSetter)} function.
71   * <p>
72   * NOTE: The batch update is not supported when {@link #keysGenerated} is in use.
73   *
74   * N.B. do not use quotes to escape the header keys. The default SQL parameter source (from Spring JDBC) can also handle
75   * headers with dotted names (e.g. <code>business.id</code>)
76   *
77   * @author Dave Syer
78   * @author Artem Bilan
79   *
80   * @since 2.0
81   */
82  public class JdbcMessageHandler extends AbstractMessageHandler {
83  
84  	private final ResultSetExtractor<List<Map<String, Object>>> generatedKeysResultSetExtractor =
85  			new RowMapperResultSetExtractor<>(new ColumnMapRowMapper(), 1);
86  
87  	private final NamedParameterJdbcOperations jdbcOperations;
88  
89  	private final String updateSql;
90  
91  	private PreparedStatementCreator generatedKeysStatementCreator;
92  
93  	private SqlParameterSourceFactory sqlParameterSourceFactory;
94  
95  	private boolean keysGenerated;
96  
97  	private MessagePreparedStatementSetter preparedStatementSetter;
98  
99  	/**
100 	 * Constructor taking {@link DataSource} from which the DB Connection can be obtained and the select query to
101 	 * execute to retrieve new rows.
102 	 * @param dataSource Must not be null
103 	 * @param updateSql query to execute
104 	 */
105 	public JdbcMessageHandler(DataSource dataSource, String updateSql) {
106 		this(new JdbcTemplate(dataSource), updateSql);
107 	}
108 
109 	/**
110 	 * Constructor taking {@link JdbcOperations} instance to use for query execution and the select query to execute to
111 	 * retrieve new rows.
112 	 * @param jdbcOperations instance to use for query execution
113 	 * @param updateSql query to execute
114 	 */
115 	public JdbcMessageHandler(JdbcOperations jdbcOperations, String updateSql) {
116 		Assert.notNull(jdbcOperations, "'jdbcOperations' must not be null.");
117 		Assert.hasText(updateSql, "'updateSql' must not be empty.");
118 		this.jdbcOperations = new NamedParameterJdbcTemplate(jdbcOperations);
119 		this.updateSql = updateSql;
120 	}
121 
122 	/**
123 	 * Flag to indicate that the update query is an insert with auto-generated keys,
124 	 * which will be logged at debug level.
125 	 * @param keysGenerated the flag value to set
126 	 */
127 	public void setKeysGenerated(boolean keysGenerated) {
128 		this.keysGenerated = keysGenerated;
129 	}
130 
131 
132 	public void setSqlParameterSourceFactory(SqlParameterSourceFactory sqlParameterSourceFactory) {
133 		this.sqlParameterSourceFactory = sqlParameterSourceFactory;
134 	}
135 
136 	/**
137 	 * Specify a {@link MessagePreparedStatementSetter} to populate parameters on the
138 	 * {@link PreparedStatement} with the {@link Message} context.
139 	 * <p>This is a low-level alternative to the {@link SqlParameterSourceFactory}.
140 	 * @param preparedStatementSetter the {@link MessagePreparedStatementSetter} to set.
141 	 * @since 4.2
142 	 */
143 	public void setPreparedStatementSetter(@Nullable MessagePreparedStatementSetter preparedStatementSetter) {
144 		this.preparedStatementSetter = preparedStatementSetter;
145 		if (preparedStatementSetter != null) {
146 			PreparedStatementCreatorFactory preparedStatementCreatorFactory =
147 					new PreparedStatementCreatorFactory(this.updateSql);
148 			preparedStatementCreatorFactory.setReturnGeneratedKeys(true);
149 			this.generatedKeysStatementCreator =
150 					preparedStatementCreatorFactory.newPreparedStatementCreator((Object[]) null);
151 		}
152 	}
153 
154 	@Override
155 	public String getComponentType() {
156 		return "jdbc:outbound-channel-adapter";
157 	}
158 
159 	@Override
160 	protected void onInit() {
161 		super.onInit();
162 		Assert.state(!(this.sqlParameterSourceFactory != null && this.preparedStatementSetter != null),
163 				"'sqlParameterSourceFactory' and 'preparedStatementSetter' are mutually exclusive.");
164 		if (this.sqlParameterSourceFactory == null && this.preparedStatementSetter == null) {
165 			this.sqlParameterSourceFactory = new BeanPropertySqlParameterSourceFactory();
166 		}
167 	}
168 
169 	/**
170 	 * Executes the update, passing the message into the {@link SqlParameterSourceFactory}.
171 	 */
172 	@Override
173 	protected void handleMessageInternal(Message<?> message) {
174 		List<? extends Map<String, Object>> keys = executeUpdateQuery(message, this.keysGenerated);
175 		if (!keys.isEmpty() && logger.isDebugEnabled()) {
176 			logger.debug("Generated keys: " + keys);
177 		}
178 	}
179 
180 	protected List<? extends Map<String, Object>> executeUpdateQuery(final Message<?> message, boolean keysGenerated) {
181 		if (keysGenerated) {
182 			if (this.preparedStatementSetter != null) {
183 				return this.jdbcOperations.getJdbcOperations()
184 						.execute(this.generatedKeysStatementCreator,
185 								ps -> {
186 									this.preparedStatementSetter.setValues(ps, message);
187 									ps.executeUpdate();
188 									ResultSet keys = ps.getGeneratedKeys(); // NOSONAR closed in JdbcUtils
189 									if (keys != null) {
190 										try {
191 
192 											return this.generatedKeysResultSetExtractor.extractData(keys);
193 										}
194 										finally {
195 											JdbcUtils.closeResultSet(keys);
196 										}
197 									}
198 									return new LinkedList<>();
199 								});
200 			}
201 			else {
202 				KeyHolder keyHolder = new GeneratedKeyHolder();
203 				this.jdbcOperations.update(this.updateSql,
204 						this.sqlParameterSourceFactory.createParameterSource(message), keyHolder);
205 				return keyHolder.getKeyList();
206 			}
207 		}
208 		else {
209 			if (message.getPayload() instanceof Iterable) {
210 				Stream<? extends Message<?>> messageStream =
211 						StreamSupport.stream(((Iterable<?>) message.getPayload()).spliterator(), false)
212 								.map(payload -> new Message<Object>() {
213 
214 									@Override
215 									public Object getPayload() {
216 										return payload;
217 									}
218 
219 									@Override
220 									public MessageHeaders getHeaders() {
221 										return message.getHeaders();
222 									}
223 
224 								});
225 
226 				int[] updates;
227 
228 				if (this.preparedStatementSetter != null) {
229 					Message<?>[] messages = messageStream.toArray(Message<?>[]::new);
230 
231 					updates = this.jdbcOperations.getJdbcOperations()
232 							.batchUpdate(this.updateSql, new BatchPreparedStatementSetter() {
233 
234 								@Override
235 								public void setValues(PreparedStatement ps, int i) throws SQLException {
236 									JdbcMessageHandler.this.preparedStatementSetter.setValues(ps, messages[i]);
237 								}
238 
239 								@Override
240 								public int getBatchSize() {
241 									return messages.length;
242 								}
243 
244 							});
245 				}
246 				else {
247 					SqlParameterSource[] sqlParameterSources =
248 							messageStream.map(this.sqlParameterSourceFactory::createParameterSource)
249 									.toArray(SqlParameterSource[]::new);
250 
251 					updates = this.jdbcOperations.batchUpdate(this.updateSql, sqlParameterSources);
252 				}
253 
254 				return Arrays.stream(updates)
255 						.mapToObj(updated -> {
256 							Map<String, Object> map = new LinkedCaseInsensitiveMap<>();
257 							map.put("UPDATED", updated);
258 							return map;
259 						})
260 						.collect(Collectors.toList());
261 			}
262 			else {
263 				int updated;
264 
265 				if (this.preparedStatementSetter != null) {
266 					updated = this.jdbcOperations.getJdbcOperations()
267 							.update(this.updateSql, ps -> this.preparedStatementSetter.setValues(ps, message));
268 				}
269 				else {
270 					updated = this.jdbcOperations.update(this.updateSql,
271 							this.sqlParameterSourceFactory.createParameterSource(message));
272 				}
273 
274 				LinkedCaseInsensitiveMap<Object> map = new LinkedCaseInsensitiveMap<>();
275 				map.put("UPDATED", updated);
276 				return Collections.singletonList(map);
277 			}
278 		}
279 	}
