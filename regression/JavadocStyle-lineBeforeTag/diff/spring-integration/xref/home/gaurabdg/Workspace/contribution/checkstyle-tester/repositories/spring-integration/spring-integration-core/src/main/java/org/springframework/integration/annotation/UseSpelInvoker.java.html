17  package org.springframework.integration.annotation;
18  
19  import java.lang.annotation.Documented;
20  import java.lang.annotation.ElementType;
21  import java.lang.annotation.Inherited;
22  import java.lang.annotation.Retention;
23  import java.lang.annotation.RetentionPolicy;
24  import java.lang.annotation.Target;
25  
26  import org.springframework.core.annotation.AliasFor;
27  
28  /**
29   * Indicates that a POJO handler method ({@code @ServiceActivator, @Transformer, } etc.,
30   * or such methods invoked from XML definitions) should be invoked using SpEL.
31   * <p>In versions prior to 5.0, such methods were always invoked using SpEL. In 5.0, the
32   * framework switched to using
33   * {@link org.springframework.messaging.handler.invocation.InvocableHandlerMethod} instead
34   * which is generally more efficient than (interpreted) SpEL.
35   * <p>There may be some unanticipated corner case where it is necessary to revert to using
36   * SpEL. Also, for very high performance requirements, you may wish to consider using
37   * compiled SpEL which is often the fastest solution (when the expression is compilable).
38   * <p>Applying this annotation to those methods will cause SpEL to be used for the
39   * invocation. An optional {@code compilerMode} property (aliased to value) is also provided.
40   *
41   * @author Gary Russell
42   * @since 5.0
43   */
44  @Target({ ElementType.TYPE, ElementType.METHOD, ElementType.ANNOTATION_TYPE })
45  @Retention(RetentionPolicy.RUNTIME)
46  @Inherited
47  @Documented
48  public @interface UseSpelInvoker {
49  
50  	/**
51  	 * Specify that the annotated method (or methods in the annotated class) will be
52  	 * invoked using SpEL instead of an
53  	 * {@link org.springframework.messaging.handler.invocation.InvocableHandlerMethod}
54  	 * with the specified compilerMode. If left empty, the default runtime compiler
55  	 * mode will be used. Must evaluate to a String containing a valid compiler mode.
56  	 * @return The compilerMode.
57  	 * @see org.springframework.expression.spel.SpelCompilerMode
58  	 */
59  	@AliasFor("compilerMode")
60  	String value() default "";
61  
62  	/**
63  	 * Specify that the annotated method (or methods in the annotated class) will be
64  	 * invoked using SpEL instead of an
65  	 * {@link org.springframework.messaging.handler.invocation.InvocableHandlerMethod}
66  	 * with the specified compilerMode. If left empty, the default runtime compiler
67  	 * mode will be used. Must evaluate to a String containing a valid compiler mode.
68  	 * @return The compilerMode.
69  	 * @see org.springframework.expression.spel.SpelCompilerMode
70  	 */
71  	@AliasFor("value")
72  	String compilerMode() default "";
73  
74  }
