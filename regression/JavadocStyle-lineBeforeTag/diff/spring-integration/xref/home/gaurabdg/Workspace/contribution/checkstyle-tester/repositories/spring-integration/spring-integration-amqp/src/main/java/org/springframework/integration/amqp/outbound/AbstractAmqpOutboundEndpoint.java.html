486 	@Override
487 	public synchronized void start() {
488 		if (!this.running) {
489 			if (!this.lazyConnect && this.connectionFactory != null) {
490 				try {
491 					Connection connection = this.connectionFactory.createConnection(); // NOSONAR (close)
492 					if (connection != null) {
493 						connection.close();
494 					}
495 				}
496 				catch (RuntimeException e) {
497 					logger.error("Failed to eagerly establish the connection.", e);
498 				}
499 			}
500 			doStart();
501 			if (this.confirmTimeout != null && getConfirmNackChannel() != null && getRabbitTemplate() != null) {
502 				this.confirmChecker = getTaskScheduler()
503 						.scheduleAtFixedRate(checkUnconfirmed(), this.confirmTimeout.dividedBy(2L));
504 			}
505 			this.running = true;
506 		}
507 	}
508 
509 	private Runnable checkUnconfirmed() {
510 		return () -> {
511 			RabbitTemplate rabbitTemplate = getRabbitTemplate();
512 			if (rabbitTemplate != null) {
513 				Collection<CorrelationData> unconfirmed =
514 						rabbitTemplate.getUnconfirmed(getConfirmTimeout().toMillis());
515 				if (unconfirmed != null) {
516 					unconfirmed.forEach(correlation -> handleConfirm(correlation, false, "Confirm timed out"));
517 				}
518 			}
519 		};
520 	}

548 	protected CorrelationData generateCorrelationData(Message<?> requestMessage) {
549 		CorrelationData correlationData = null;
550 		if (this.correlationDataGenerator != null) {
551 			UUID messageId = requestMessage.getHeaders().getId();
552 			if (messageId == null) {
553 				messageId = NO_ID;
554 			}
555 			Object userData = this.correlationDataGenerator.processMessage(requestMessage);
556 			if (userData != null) {
557 				correlationData = new CorrelationDataWrapper(messageId.toString(), userData, requestMessage);
558 			}
559 			else {
560 				this.logger.debug("'confirmCorrelationExpression' resolved to 'null'; "
561 						+ "no publisher confirm will be sent to the ack or nack channel");
562 			}
563 		}
564 		return correlationData;
565 	}

589 	protected AbstractIntegrationMessageBuilder<?> buildReply(MessageConverter converter,
590 			org.springframework.amqp.core.Message amqpReplyMessage) {
591 
592 		Object replyObject = converter.fromMessage(amqpReplyMessage);
593 		AbstractIntegrationMessageBuilder<?> builder = prepareMessageBuilder(replyObject);
594 		Map<String, ?> headers = getHeaderMapper().toHeadersFromReply(amqpReplyMessage.getMessageProperties());
595 		builder.copyHeadersIfAbsent(headers);
596 		return builder;
597 	}

599 	private AbstractIntegrationMessageBuilder<?> prepareMessageBuilder(Object replyObject) {
600 		return replyObject instanceof Message
601 				? getMessageBuilderFactory().fromMessage((Message<?>) replyObject)
602 				: getMessageBuilderFactory().withPayload(replyObject);
603 	}

605 	protected Message<?> buildReturnedMessage(org.springframework.amqp.core.Message message,
606 			int replyCode, String replyText, String exchange, String returnedRoutingKey, MessageConverter converter) {
607 
608 		Object returnedObject = converter.fromMessage(message);
609 		AbstractIntegrationMessageBuilder<?> builder = prepareMessageBuilder(returnedObject);
610 		Map<String, ?> headers = getHeaderMapper().toHeadersFromReply(message.getMessageProperties());
611 		if (this.errorMessageStrategy == null) {
612 			builder.copyHeadersIfAbsent(headers)
613 					.setHeader(AmqpHeaders.RETURN_REPLY_CODE, replyCode)
614 					.setHeader(AmqpHeaders.RETURN_REPLY_TEXT, replyText)
615 					.setHeader(AmqpHeaders.RETURN_EXCHANGE, exchange)
616 					.setHeader(AmqpHeaders.RETURN_ROUTING_KEY, returnedRoutingKey);
617 		}
618 		Message<?> returnedMessage = builder.build();
619 		if (this.errorMessageStrategy != null) {
620 			returnedMessage = this.errorMessageStrategy.buildErrorMessage(new ReturnedAmqpMessageException(
621 					returnedMessage, message, replyCode, replyText, exchange, returnedRoutingKey), null);
622 		}
623 		return returnedMessage;
624 	}

626 	protected void handleConfirm(CorrelationData correlationData, boolean ack, String cause) {
627 		CorrelationDataWrapper wrapper = (CorrelationDataWrapper) correlationData;
628 		if (correlationData == null) {
629 			if (logger.isDebugEnabled()) {
630 				logger.debug("No correlation data provided for ack: " + ack + " cause:" + cause);
631 			}
632 			return;
633 		}
634 		Object userCorrelationData = wrapper.getUserData();
635 		Message<?> confirmMessage;
636 		confirmMessage = buildConfirmMessage(ack, cause, wrapper, userCorrelationData);
637 		if (ack && getConfirmAckChannel() != null) {
638 			sendOutput(confirmMessage, getConfirmAckChannel(), true);
639 		}
640 		else if (!ack && getConfirmNackChannel() != null) {
641 			sendOutput(confirmMessage, getConfirmNackChannel(), true);
642 		}
643 		else {
644 			if (logger.isInfoEnabled()) {
645 				logger.info("Nowhere to send publisher confirm "
646 						+ (ack ? "ack" : "nack") + " for "
647 						+ userCorrelationData);
648 			}
649 		}
650 	}
651 
652 	private Message<?> buildConfirmMessage(boolean ack, String cause, CorrelationDataWrapper wrapper,
653 			Object userCorrelationData) {
654 
655 		if (this.errorMessageStrategy == null || ack) {
656 			Map<String, Object> headers = new HashMap<>();
657 			headers.put(AmqpHeaders.PUBLISH_CONFIRM, ack);
658 			if (!ack && StringUtils.hasText(cause)) {
659 				headers.put(AmqpHeaders.PUBLISH_CONFIRM_NACK_CAUSE, cause);
660 			}
661 
662 			return prepareMessageBuilder(userCorrelationData)
663 					.copyHeaders(headers)
664 					.build();
665 		}
666 		else {
667 			return this.errorMessageStrategy.buildErrorMessage(new NackedAmqpMessageException(wrapper.getMessage(),
668 					wrapper.getUserData(), cause), null);
669 		}
670 	}
