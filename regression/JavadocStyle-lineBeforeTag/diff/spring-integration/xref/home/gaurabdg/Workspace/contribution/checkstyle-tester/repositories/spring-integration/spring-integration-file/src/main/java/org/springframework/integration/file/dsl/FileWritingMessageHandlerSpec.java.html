17  package org.springframework.integration.file.dsl;
18  
19  import java.io.File;
20  import java.util.Collections;
21  import java.util.Map;
22  import java.util.function.Function;
23  
24  import org.springframework.expression.Expression;
25  import org.springframework.integration.dsl.ComponentsRegistration;
26  import org.springframework.integration.dsl.MessageHandlerSpec;
27  import org.springframework.integration.expression.FunctionExpression;
28  import org.springframework.integration.file.DefaultFileNameGenerator;
29  import org.springframework.integration.file.FileNameGenerator;
30  import org.springframework.integration.file.FileWritingMessageHandler;
31  import org.springframework.integration.file.support.FileExistsMode;
32  import org.springframework.messaging.Message;
33  import org.springframework.scheduling.TaskScheduler;
34  import org.springframework.util.Assert;
35  
36  /**
37   * The {@link MessageHandlerSpec} for the {@link FileWritingMessageHandler}.
38   *
39   * @author Artem Bilan
40   * @author Gary Russell
41   *
42   * @since 5.0
43   */
44  public class FileWritingMessageHandlerSpec
45  		extends MessageHandlerSpec<FileWritingMessageHandlerSpec, FileWritingMessageHandler>
46  		implements ComponentsRegistration {
47  
48  	private FileNameGenerator fileNameGenerator;
49  
50  	private DefaultFileNameGenerator defaultFileNameGenerator;
51  
52  	protected FileWritingMessageHandlerSpec(File destinationDirectory) {
53  		this.target = new FileWritingMessageHandler(destinationDirectory);
54  	}
55  
56  	protected FileWritingMessageHandlerSpec(String directoryExpression) {
57  		this(PARSER.parseExpression(directoryExpression));
58  	}
59  
60  	protected <P> FileWritingMessageHandlerSpec(Function<Message<P>, ?> directoryFunction) {
61  		this(new FunctionExpression<>(directoryFunction));
62  	}
63  
64  	protected FileWritingMessageHandlerSpec(Expression directoryExpression) {
65  		this.target = new FileWritingMessageHandler(directoryExpression);
66  	}
67  
68  	FileWritingMessageHandlerSpec expectReply(boolean expectReply) {
69  		this.target.setExpectReply(expectReply);
70  		if (expectReply) {
71  			this.target.setRequiresReply(true);
72  		}
73  		return _this();
74  	}
75  
76  	/**
77  	 * Specify whether to create the destination directory automatically if it
78  	 * does not yet exist upon initialization. By default, this value is
79  	 * <em>true</em>. If set to <em>false</em> and the
80  	 * destination directory does not exist, an Exception will be thrown upon
81  	 * initialization.
82  	 * @param autoCreateDirectory true to create the directory if needed.
83  	 * @return the current Spec
84  	 */
85  	public FileWritingMessageHandlerSpec autoCreateDirectory(boolean autoCreateDirectory) {
86  		this.target.setAutoCreateDirectory(autoCreateDirectory);
87  		return _this();
88  	}
89  
90  	/**
91  	 * By default, every file that is in the process of being transferred will
92  	 * appear in the file system with an additional suffix, which by default is {@code .writing}.
93  	 * @param temporaryFileSuffix The temporary file suffix.
94  	 * @return the current Spec
95  	 */
96  	public FileWritingMessageHandlerSpec temporaryFileSuffix(String temporaryFileSuffix) {
97  		this.target.setTemporaryFileSuffix(temporaryFileSuffix);
98  		return _this();
99  	}
100 
101 	/**
102 	 * Set the {@link FileExistsMode} that specifies what will happen in
103 	 * case the destination exists.
104 	 * @param fileExistsMode the {@link FileExistsMode} to consult.
105 	 * @return the current Spec
106 	 */
107 	public FileWritingMessageHandlerSpec fileExistsMode(FileExistsMode fileExistsMode) {
108 		this.target.setFileExistsMode(fileExistsMode);
109 		return _this();
110 	}
111 
112 	/**
113 	 * Set the file name generator used to generate the target file name.
114 	 * Default {@link DefaultFileNameGenerator}.
115 	 * @param fileNameGenerator the file name generator.
116 	 * @return the current Spec
117 	 */
118 	public FileWritingMessageHandlerSpec fileNameGenerator(FileNameGenerator fileNameGenerator) {
119 		this.fileNameGenerator = fileNameGenerator;
120 		this.target.setFileNameGenerator(fileNameGenerator);
121 		return _this();
122 	}
123 
124 	/**
125 	 * Set the {@link DefaultFileNameGenerator} based on the provided SpEL expression.
126 	 * @param fileNameExpression the SpEL expression for file names generation.
127 	 * @return the current Spec
128 	 */
129 	public FileWritingMessageHandlerSpec fileNameExpression(String fileNameExpression) {
130 		Assert.isNull(this.fileNameGenerator,
131 				"'fileNameGenerator' and 'fileNameGeneratorExpression' are mutually exclusive.");
132 		this.defaultFileNameGenerator = new DefaultFileNameGenerator();
133 		this.defaultFileNameGenerator.setExpression(fileNameExpression);
134 		return fileNameGenerator(this.defaultFileNameGenerator);
135 	}
136 
137 	/**
138 	 * Specify whether to delete source Files after writing to the destination directory.
139 	 * The default is <em>false</em>. When set to <em>true</em>, it will only have an
140 	 * effect if the inbound Message has a File payload or a
141 	 * {@link org.springframework.integration.file.FileHeaders#ORIGINAL_FILE} header value
142 	 * containing either a File instance or a String representing the original file path.
143 	 * @param deleteSourceFiles true to delete the source files.
144 	 * @return the current Spec
145 	 */
146 	public FileWritingMessageHandlerSpec deleteSourceFiles(boolean deleteSourceFiles) {
147 		this.target.setDeleteSourceFiles(deleteSourceFiles);
148 		return _this();
149 	}
150 
151 	/**
152 	 * Set the charset to use when converting String payloads to bytes as the content of the file.
153 	 * Default {@code UTF-8}.
154 	 * @param charset the charset.
155 	 * @return the current Spec
156 	 */
157 	public FileWritingMessageHandlerSpec charset(String charset) {
158 		this.target.setCharset(charset);
159 		return _this();
160 	}
161 
162 	/**
163 	 * If {@code true} will append a new-line after each write.
164 	 * Defaults to {@code false}.
165 	 * @param appendNewLine true if a new-line should be written to the file after payload is written.
166 	 * @return the spec.
167 	 * @see FileWritingMessageHandler#setAppendNewLine(boolean)
168 	 */
169 	public FileWritingMessageHandlerSpec appendNewLine(boolean appendNewLine) {
170 		this.target.setAppendNewLine(appendNewLine);
171 		return this;
172 	}
173 
174 	/**
175 	 * Set the buffer size to use while writing to files; default 8192.
176 	 * @param bufferSize the buffer size.
177 	 * @return the spec.
178 	 * @see FileWritingMessageHandler#setBufferSize(int)
179 	 */
180 	public FileWritingMessageHandlerSpec bufferSize(int bufferSize) {
181 		this.target.setBufferSize(bufferSize);
182 		return this;
183 	}
184 
185 	/**
186 	 * Set the frequency to flush buffers when {@link FileExistsMode#APPEND_NO_FLUSH} is
187 	 * being used.
188 	 * @param flushInterval the interval.
189 	 * @return the spec.
190 	 * @see FileWritingMessageHandler#setBufferSize(int)
191 	 * @see #flushWhenIdle(boolean)
192 	 */
193 	public FileWritingMessageHandlerSpec flushInterval(long flushInterval) {
194 		this.target.setFlushInterval(flushInterval);
195 		return this;
196 	}
197 
198 	/**
199 	 * Set the flush when idle flag to false if you wish the interval to apply to when
200 	 * the file was opened rather than when the file was last written.
201 	 * @param flushWhenIdle false to flush if the interval since the file was opened has elapsed.
202 	 * @return the spec.
203 	 * @see FileWritingMessageHandler#setFlushWhenIdle(boolean)
204 	 * @see #flushInterval(long)
205 	 */
206 	public FileWritingMessageHandlerSpec flushWhenIdle(boolean flushWhenIdle) {
207 		this.target.setFlushWhenIdle(flushWhenIdle);
208 		return this;
209 	}
210 
211 	/**
212 	 * Specify a {@link TaskScheduler} for flush task when the {@link FileExistsMode#APPEND_NO_FLUSH} is in use.
213 	 * @param taskScheduler the {@link TaskScheduler} to use.
214 	 * @return the spec.
215 	 * @see FileWritingMessageHandler#setTaskScheduler(TaskScheduler)
216 	 */
217 	public FileWritingMessageHandlerSpec taskScheduler(TaskScheduler taskScheduler) {
218 		this.target.setTaskScheduler(taskScheduler);
219 		return this;
220 	}
221 
222 	/**
223 	 * Specify a {@link FileWritingMessageHandler.MessageFlushPredicate} for flush task
224 	 * when the {@link FileExistsMode#APPEND_NO_FLUSH} is in use.
225 	 * @param flushPredicate the {@link FileWritingMessageHandler.MessageFlushPredicate} to use.
226 	 * @return the spec.
227 	 * @see FileWritingMessageHandler#setFlushPredicate(FileWritingMessageHandler.MessageFlushPredicate)
228 	 */
229 	public FileWritingMessageHandlerSpec flushPredicate(
230 			FileWritingMessageHandler.MessageFlushPredicate flushPredicate) {
231 		this.target.setFlushPredicate(flushPredicate);
232 		return this;
233 	}
234 
235 	/**
236 	 * Set to true to preserve the destination file timestamp. If true and
237 	 * the payload is a {@link File}, the payload's {@code lastModified} time will be
238 	 * transferred to the destination file.
239 	 * @param preserveTimestamp the {@code boolean} flag to use.
240 	 * @return the spec.
241 	 * @see FileWritingMessageHandler#setPreserveTimestamp(boolean)
242 	 */
243 	public FileWritingMessageHandlerSpec preserveTimestamp(boolean preserveTimestamp) {
244 		this.target.setPreserveTimestamp(preserveTimestamp);
245 		return this;
246 	}
247 
248 	/**
249 	 * Set the file permissions after uploading, e.g. 0600 for
250 	 * owner read/write. Only applies to file systems that support posix
251 	 * file permissions.
252 	 * @param chmod the permissions.
253 	 * @throws IllegalArgumentException if the value is higher than 0777.
254 	 * @return the spec.
255 	 * @see FileWritingMessageHandler#setChmod(int)
256 	 */
257 	public FileWritingMessageHandlerSpec chmod(int chmod) {
258 		this.target.setChmod(chmod);
259 		return this;
260 	}
261 
262 	@Override
263 	public Map<Object, String> getComponentsToRegister() {
264 		if (this.defaultFileNameGenerator != null) {
265 			return Collections.singletonMap(this.defaultFileNameGenerator, null);
266 		}
267 		return null;
268 	}

270 }
