17  package org.springframework.integration.mail.config;
18  
19  import static org.assertj.core.api.Assertions.assertThat;
20  
21  import java.util.Properties;
22  
23  import javax.mail.URLName;
24  
25  import org.junit.Test;
26  import org.junit.runner.RunWith;
27  
28  import org.springframework.beans.DirectFieldAccessor;
29  import org.springframework.beans.factory.annotation.Autowired;
30  import org.springframework.context.ApplicationContext;
31  import org.springframework.integration.endpoint.SourcePollingChannelAdapter;
32  import org.springframework.integration.mail.ImapMailReceiver;
33  import org.springframework.integration.mail.MailReceivingMessageSource;
34  import org.springframework.integration.mail.Pop3MailReceiver;
35  import org.springframework.test.context.ContextConfiguration;
36  import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
37  
38  /**
39   * @author Jonas Partner
40   * @author Mark Fisher
41   */
42  @RunWith(SpringJUnit4ClassRunner.class)
43  @ContextConfiguration
44  public class PollingMailSourceParserTests {
45  
46  	@Autowired
47  	private ApplicationContext context;
48  
49  
50  	@Test
51  	public void imapAdapter() {
52  		Object adapter = context.getBean("imapAdapter");
53  		assertThat(adapter.getClass()).isEqualTo(SourcePollingChannelAdapter.class);
54  		DirectFieldAccessor adapterAccessor = new DirectFieldAccessor(adapter);
55  		assertThat(adapterAccessor.getPropertyValue("autoStartup")).isEqualTo(Boolean.FALSE);
56  		Object channel = context.getBean("channel");
57  		assertThat(adapterAccessor.getPropertyValue("outputChannel")).isEqualTo(channel);
58  		Object source = adapterAccessor.getPropertyValue("source");
59  		assertThat(source.getClass()).isEqualTo(MailReceivingMessageSource.class);
60  		Object receiver = new DirectFieldAccessor(source).getPropertyValue("mailReceiver");
61  		assertThat(receiver.getClass()).isEqualTo(ImapMailReceiver.class);
62  		DirectFieldAccessor receiverAccessor = new DirectFieldAccessor(receiver);
63  		Object url = receiverAccessor.getPropertyValue("url");
64  		assertThat(url).isEqualTo(new URLName("imap:foo"));
65  		Properties properties = (Properties) receiverAccessor.getPropertyValue("javaMailProperties");
66  		assertThat(properties.getProperty("foo")).isEqualTo("bar");
67  	}
68  
69  	@Test
70  	public void pop3Adapter() {
71  		Object adapter = context.getBean("pop3Adapter");
72  		assertThat(adapter.getClass()).isEqualTo(SourcePollingChannelAdapter.class);
73  		DirectFieldAccessor adapterAccessor = new DirectFieldAccessor(adapter);
74  		assertThat(adapterAccessor.getPropertyValue("autoStartup")).isEqualTo(Boolean.FALSE);
75  		Object channel = context.getBean("channel");
76  		assertThat(adapterAccessor.getPropertyValue("outputChannel")).isEqualTo(channel);
77  		Object source = adapterAccessor.getPropertyValue("source");
78  		assertThat(source.getClass()).isEqualTo(MailReceivingMessageSource.class);
79  		Object receiver = new DirectFieldAccessor(source).getPropertyValue("mailReceiver");
80  		assertThat(receiver.getClass()).isEqualTo(Pop3MailReceiver.class);
81  		DirectFieldAccessor receiverAccessor = new DirectFieldAccessor(receiver);
82  		Object url = receiverAccessor.getPropertyValue("url");
83  		assertThat(url).isEqualTo(new URLName("pop3:bar"));
84  		Properties properties = (Properties) receiverAccessor.getPropertyValue("javaMailProperties");
85  		assertThat(properties.getProperty("foo")).isEqualTo("bar");
86  	}
