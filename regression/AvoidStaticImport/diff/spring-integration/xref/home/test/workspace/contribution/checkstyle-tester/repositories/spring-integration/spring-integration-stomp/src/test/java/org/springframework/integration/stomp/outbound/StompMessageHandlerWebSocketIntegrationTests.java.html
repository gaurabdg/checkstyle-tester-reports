89  @SpringJUnitConfig(StompMessageHandlerWebSocketIntegrationTests.ContextConfiguration.class)
90  @LongRunningTest
91  @DirtiesContext
92  public class StompMessageHandlerWebSocketIntegrationTests {
93  
94  	@Value("#{server.serverContext}")
95  	private ApplicationContext serverContext;
96  
97  	@Autowired
98  	@Qualifier("webSocketOutputChannel")
99  	private MessageChannel webSocketOutputChannel;
100 
101 	@Autowired
102 	@Qualifier("stompEvents")
103 	private PollableChannel stompEvents;
104 
105 	@Test
106 	public void testStompMessageHandler() throws InterruptedException {
107 		StompHeaderAccessor headers = StompHeaderAccessor.create(StompCommand.SEND);
108 		headers.setDestination("/app/simple");
109 		Message<String> message = MessageBuilder.withPayload("foo").setHeaders(headers).build();
110 		this.webSocketOutputChannel.send(message);
111 
112 		SimpleController controller = this.serverContext.getBean(SimpleController.class);
113 		assertThat(controller.latch.await(10, TimeUnit.SECONDS)).isTrue();
114 
115 		Message<?> receive = this.stompEvents.receive(10000);
116 		assertThat(receive).isNotNull();
117 		assertThat(receive.getPayload()).isInstanceOf(StompSessionConnectedEvent.class);
118 
119 		// Simple Broker Relay doesn't support RECEIPT Frame, so we check here the 'lost' StompReceiptEvent
120 		receive = this.stompEvents.receive(10000);
121 		assertThat(receive).isNotNull();
122 		assertThat(receive.getPayload()).isInstanceOf(StompReceiptEvent.class);
123 		StompReceiptEvent stompReceiptEvent = (StompReceiptEvent) receive.getPayload();
124 		assertThat(stompReceiptEvent.getStompCommand()).isEqualTo(StompCommand.SEND);
125 		assertThat(stompReceiptEvent.getDestination()).isEqualTo("/app/simple");
126 		assertThat(stompReceiptEvent.isLost()).isTrue();
127 		assertThat(stompReceiptEvent.getMessage()).isNotNull();
128 
129 		headers = StompHeaderAccessor.create(StompCommand.SEND);
130 		headers.setDestination("/foo");
131 		message = MessageBuilder.withPayload("bar").setHeaders(headers).build();
132 		this.webSocketOutputChannel.send(message);
133 
134 		receive = this.stompEvents.receive(10000);
135 		assertThat(receive).isNotNull();
136 		assertThat(receive.getPayload()).isInstanceOf(StompExceptionEvent.class);
137 		StompExceptionEvent stompExceptionEvent = (StompExceptionEvent) receive.getPayload();
138 		Throwable cause = stompExceptionEvent.getCause();
139 		assertThat(cause).isInstanceOf(MessageDeliveryException.class);
140 		MessageDeliveryException messageDeliveryException = (MessageDeliveryException) cause;
141 		Message<?> failedMessage = messageDeliveryException.getFailedMessage();
142 		assertThat((String) failedMessage.getPayload()).contains("preSend intentional Exception");
143 
144 		receive = this.stompEvents.receive(10000);
145 		assertThat(receive).isNotNull();
146 		assertThat(receive.getPayload()).isInstanceOf(StompConnectionFailedEvent.class);
147 
148 		receive = this.stompEvents.receive(10000);
149 		assertThat(receive).isNotNull();
150 		assertThat(receive.getPayload()).isInstanceOf(StompSessionConnectedEvent.class);
151 
152 		receive = this.stompEvents.receive(10000);
153 		assertThat(receive).isNotNull();
154 		assertThat(receive.getPayload()).isInstanceOf(StompReceiptEvent.class);
155 		stompReceiptEvent = (StompReceiptEvent) receive.getPayload();
156 		assertThat(stompReceiptEvent.getStompCommand()).isEqualTo(StompCommand.SEND);
157 		assertThat(stompReceiptEvent.getDestination()).isEqualTo("/foo");
158 		assertThat(stompReceiptEvent.isLost()).isTrue();
159 	}

163 	@Configuration
164 	@EnableIntegration
165 	public static class ContextConfiguration {
166 
167 		@Bean
168 		public TomcatWebSocketTestServer server() {
169 			return new TomcatWebSocketTestServer(ServerConfig.class);
170 		}
171 
172 		@Bean
173 		public WebSocketClient webSocketClient() {
174 			return new SockJsClient(Collections.<Transport>singletonList(new WebSocketTransport(new StandardWebSocketClient())));
175 		}
176 
177 		@Bean
178 		public WebSocketStompClient stompClient(TaskScheduler taskScheduler) {
179 			WebSocketStompClient webSocketStompClient = new WebSocketStompClient(webSocketClient());
180 			webSocketStompClient.setTaskScheduler(taskScheduler);
181 			webSocketStompClient.setReceiptTimeLimit(5000);
182 			webSocketStompClient.setMessageConverter(new StringMessageConverter());
183 			return webSocketStompClient;
184 		}
185 
186 		@Bean
187 		public StompSessionManager stompSessionManager(WebSocketStompClient stompClient) {
188 			WebSocketStompSessionManager webSocketStompSessionManager =
189 					new WebSocketStompSessionManager(stompClient, server().getWsBaseUrl() + "/ws");
190 			webSocketStompSessionManager.setAutoReceipt(true);
191 			webSocketStompSessionManager.setRecoveryInterval(1000);
192 			return webSocketStompSessionManager;
193 		}
194 
195 		@Bean
196 		@ServiceActivator(inputChannel = "webSocketOutputChannel")
197 		public MessageHandler stompMessageHandler(StompSessionManager stompSessionManager) {
198 			StompMessageHandler stompMessageHandler = new StompMessageHandler(stompSessionManager);
199 			stompMessageHandler.setConnectTimeout(10000);
200 			return stompMessageHandler;
201 		}
202 
203 		@Bean
204 		public PollableChannel stompEvents() {
205 			return new QueueChannel();
206 		}
207 
208 		@Bean
209 		public ApplicationListener<ApplicationEvent> stompEventListener() {
210 			ApplicationEventListeningMessageProducer producer = new ApplicationEventListeningMessageProducer();
211 			producer.setEventTypes(StompIntegrationEvent.class);
212 			producer.setOutputChannel(stompEvents());
213 			return producer;
214 		}

233 		public void handle() {
234 			this.latch.countDown();
235 		}

239 	@Configuration
240 	@EnableWebSocketMessageBroker
241 	@ComponentScan(
242 			basePackageClasses = StompMessageHandlerWebSocketIntegrationTests.class,
243 			useDefaultFilters = false,
244 			includeFilters = @ComponentScan.Filter(IntegrationTestController.class))
245 	static class ServerConfig implements WebSocketMessageBrokerConfigurer {
246 
247 		@Bean
248 		public DefaultHandshakeHandler handshakeHandler() {
249 			return new DefaultHandshakeHandler(new TomcatRequestUpgradeStrategy());
250 		}
251 
252 		@Override
253 		public void registerStompEndpoints(StompEndpointRegistry registry) {
254 			registry.addEndpoint("/ws").setHandshakeHandler(handshakeHandler()).withSockJS();
255 		}
256 
257 		@Override
258 		public void configureMessageBroker(MessageBrokerRegistry configurer) {
259 			configurer.setApplicationDestinationPrefixes("/app");
260 			configurer.enableSimpleBroker("/topic", "/queue");
261 		}
262 
263 
264 		@Override
265 		public void configureClientInboundChannel(ChannelRegistration registration) {
266 			registration.interceptors(new ChannelInterceptor() {
267 
268 				private final AtomicBoolean invoked = new AtomicBoolean();
269 
270 				@Override
271 				public Message<?> preSend(Message<?> message, MessageChannel channel) {
272 					if (StompCommand.CONNECT.equals(StompHeaderAccessor.wrap(message).getCommand()) ||
273 							this.invoked.compareAndSet(false, true)) {
274 						return message;
275 					}
276 					throw new RuntimeException("preSend intentional Exception");
277 				}
278 
279 			});
280 		}
