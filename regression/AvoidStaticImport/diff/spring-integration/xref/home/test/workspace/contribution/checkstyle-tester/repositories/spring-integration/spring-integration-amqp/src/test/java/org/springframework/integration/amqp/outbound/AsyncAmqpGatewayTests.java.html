67  @RabbitAvailable(queues = { "asyncQ1", "asyncRQ1" })
68  @LogLevels(categories = "org.springframework.amqp", level = "trace")
69  class AsyncAmqpGatewayTests {
70  
71  	@Test
72  	void testConfirmsAndReturns() throws Exception {
73  		CachingConnectionFactory ccf = new CachingConnectionFactory("localhost");
74  		ccf.setPublisherConfirmType(CachingConnectionFactory.ConfirmType.CORRELATED);
75  		ccf.setPublisherReturns(true);
76  		RabbitTemplate template = new RabbitTemplate(ccf);
77  		SimpleMessageListenerContainer container = new SimpleMessageListenerContainer(ccf);
78  		container.setBeanName("replyContainer");
79  		container.setQueueNames("asyncRQ1");
80  		container.afterPropertiesSet();
81  		container.start();
82  		AsyncRabbitTemplate asyncTemplate = new AsyncRabbitTemplate(template, container);
83  		asyncTemplate.setEnableConfirms(true);
84  		asyncTemplate.setMandatory(true);
85  
86  		SimpleMessageListenerContainer receiver = new SimpleMessageListenerContainer(ccf);
87  		receiver.setBeanName("receiver");
88  		receiver.setQueueNames("asyncQ1");
89  		final CountDownLatch waitForAckBeforeReplying = new CountDownLatch(1);
90  		MessageListenerAdapter messageListener = new MessageListenerAdapter(
91  				(ReplyingMessageListener<String, String>) foo -> {
92  					try {
93  						waitForAckBeforeReplying.await(10, TimeUnit.SECONDS);
94  					}
95  					catch (@SuppressWarnings("unused") InterruptedException e) {
96  						Thread.currentThread().interrupt();
97  					}
98  					return foo.toUpperCase();
99  				});
100 		receiver.setMessageListener(messageListener);
101 		receiver.afterPropertiesSet();
102 		receiver.start();
103 
104 		AsyncAmqpOutboundGateway gateway = new AsyncAmqpOutboundGateway(asyncTemplate);
105 		Log logger = spy(TestUtils.getPropertyValue(gateway, "logger", Log.class));
106 		given(logger.isDebugEnabled()).willReturn(true);
107 		final CountDownLatch replyTimeoutLatch = new CountDownLatch(1);
108 		willAnswer(invocation -> {
109 			invocation.callRealMethod();
110 			replyTimeoutLatch.countDown();
111 			return null;
112 		}).given(logger).debug(startsWith("Reply not required and async timeout for"));
113 		new DirectFieldAccessor(gateway).setPropertyValue("logger", logger);
114 		QueueChannel outputChannel = new QueueChannel();
115 		outputChannel.setBeanName("output");
116 		QueueChannel returnChannel = new QueueChannel();
117 		returnChannel.setBeanName("returns");
118 		QueueChannel ackChannel = new QueueChannel();
119 		ackChannel.setBeanName("acks");
120 		QueueChannel errorChannel = new QueueChannel();
121 		errorChannel.setBeanName("errors");
122 		gateway.setOutputChannel(outputChannel);
123 		gateway.setReturnChannel(returnChannel);
124 		gateway.setConfirmAckChannel(ackChannel);
125 		gateway.setConfirmNackChannel(ackChannel);
126 		gateway.setConfirmCorrelationExpressionString("#this");
127 		gateway.setExchangeName("");
128 		gateway.setRoutingKey("asyncQ1");
129 		gateway.setBeanFactory(mock(BeanFactory.class));
130 		gateway.afterPropertiesSet();
131 		gateway.start();
132 
133 		Message<?> message = MessageBuilder.withPayload("foo").setErrorChannel(errorChannel).build();
134 
135 		gateway.handleMessage(message);
136 
137 		Message<?> ack = ackChannel.receive(10000);
138 		assertThat(ack).isNotNull();
139 		assertThat(ack.getPayload()).isEqualTo("foo");
140 		assertThat(ack.getHeaders().get(AmqpHeaders.PUBLISH_CONFIRM)).isEqualTo(true);
141 		waitForAckBeforeReplying.countDown();
142 
143 		Message<?> received = outputChannel.receive(10000);
144 		assertThat(received).isNotNull();
145 		assertThat(received.getPayload()).isEqualTo("FOO");
146 
147 		// timeout tests
148 		asyncTemplate.setReceiveTimeout(10);
149 
150 		receiver.setMessageListener(message1 -> { });
151 		// reply timeout with no requiresReply
152 		message = MessageBuilder.withPayload("bar").setErrorChannel(errorChannel).build();
153 		gateway.handleMessage(message);
154 		assertThat(replyTimeoutLatch.await(10, TimeUnit.SECONDS)).isTrue();
155 
156 		// reply timeout with requiresReply
157 		gateway.setRequiresReply(true);
158 		message = MessageBuilder.withPayload("baz").setErrorChannel(errorChannel).build();
159 		gateway.handleMessage(message);
160 
161 		received = errorChannel.receive(10000);
162 		assertThat(received).isInstanceOf(ErrorMessage.class);
163 		ErrorMessage error = (ErrorMessage) received;
164 		assertThat(error.getPayload()).isInstanceOf(MessagingException.class);
165 		assertThat(error.getPayload().getCause()).isInstanceOf(AmqpReplyTimeoutException.class);
166 		asyncTemplate.setReceiveTimeout(30000);
167 		receiver.setMessageListener(messageListener);
168 
169 		// error on sending result
170 		DirectChannel errorForce = new DirectChannel();
171 		errorForce.setBeanName("errorForce");
172 		errorForce.subscribe(message1 -> {
173 			throw new RuntimeException("intentional");
174 		});
175 		gateway.setOutputChannel(errorForce);
176 		message = MessageBuilder.withPayload("qux").setErrorChannel(errorChannel).build();
177 		gateway.handleMessage(message);
178 		received = errorChannel.receive(10000);
179 		assertThat(received).isInstanceOf(ErrorMessage.class);
180 		error = (ErrorMessage) received;
181 		assertThat(error.getPayload()).isInstanceOf(MessagingException.class);
182 		assertThat(((MessagingException) error.getPayload()).getFailedMessage().getPayload()).isEqualTo("QUX");
183 
184 		gateway.setRoutingKey(UUID.randomUUID().toString());
185 		message = MessageBuilder.withPayload("fiz").setErrorChannel(errorChannel).build();
186 		gateway.handleMessage(message);
187 		Message<?> returned = returnChannel.receive(10000);
188 		assertThat(returned).isNotNull();
189 		assertThat(returned).isInstanceOf(ErrorMessage.class);
190 		assertThat(returned.getPayload()).isInstanceOf(ReturnedAmqpMessageException.class);
191 		ReturnedAmqpMessageException payload = (ReturnedAmqpMessageException) returned.getPayload();
192 		assertThat(payload.getFailedMessage().getPayload()).isEqualTo("fiz");
193 		ackChannel.receive(10000);
194 		ackChannel.purge(null);
195 
196 		asyncTemplate = mock(AsyncRabbitTemplate.class);
197 		RabbitMessageFuture future = asyncTemplate.new RabbitMessageFuture(null, null);
198 		willReturn(future).given(asyncTemplate).sendAndReceive(anyString(), anyString(),
199 				any(org.springframework.amqp.core.Message.class));
200 		DirectFieldAccessor dfa = new DirectFieldAccessor(future);
201 		dfa.setPropertyValue("nackCause", "nacknack");
202 		SettableListenableFuture<Boolean> confirmFuture = new SettableListenableFuture<>();
203 		confirmFuture.set(false);
204 		dfa.setPropertyValue("confirm", confirmFuture);
205 		new DirectFieldAccessor(gateway).setPropertyValue("template", asyncTemplate);
206 
207 		message = MessageBuilder.withPayload("buz").setErrorChannel(errorChannel).build();
208 		gateway.handleMessage(message);
209 
210 		ack = ackChannel.receive(10000);
211 		assertThat(ack).isNotNull();
212 		assertThat(returned).isInstanceOf(ErrorMessage.class);
213 		assertThat(returned.getPayload()).isInstanceOf(ReturnedAmqpMessageException.class);
214 		NackedAmqpMessageException nack = (NackedAmqpMessageException) ack.getPayload();
215 		assertThat(nack.getFailedMessage().getPayload()).isEqualTo("buz");
216 		assertThat(nack.getNackReason()).isEqualTo("nacknack");
217 
218 		asyncTemplate.stop();
219 		receiver.stop();
220 		ccf.destroy();
221 	}
