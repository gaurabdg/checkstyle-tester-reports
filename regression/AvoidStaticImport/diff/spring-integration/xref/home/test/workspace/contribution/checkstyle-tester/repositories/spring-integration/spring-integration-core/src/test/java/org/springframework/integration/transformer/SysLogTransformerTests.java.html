35  	private final SyslogToMapTransformer sut = new SyslogToMapTransformer();
36  
37  	@Test
38  	public void testMap() throws Exception {
39  		String syslog = "<158>JUL 26 22:08:35 WEBERN TESTING[70729]: TEST SYSLOG MESSAGE";
40  		Map<String, ?> transformed = sut.transformPayload(syslog.getBytes());
41  		assertThat(transformed.size()).isEqualTo(6);
42  		assertThat(transformed.get(SyslogToMapTransformer.FACILITY)).isEqualTo(19);
43  		assertThat(transformed.get(SyslogToMapTransformer.SEVERITY)).isEqualTo(6);
44  		Object date = transformed.get(SyslogToMapTransformer.TIMESTAMP);
45  		assertThat(date instanceof Date || date instanceof String).isTrue();
46  		assertThat(transformed.get(SyslogToMapTransformer.HOST)).isEqualTo("WEBERN");
47  		assertThat(transformed.get(SyslogToMapTransformer.TAG)).isEqualTo("TESTING");
48  		assertThat(transformed.get(SyslogToMapTransformer.MESSAGE)).isEqualTo("[70729]: TEST SYSLOG MESSAGE");
49  
50  		String[] fields = { SyslogToMapTransformer.FACILITY, SyslogToMapTransformer.SEVERITY,
51  				SyslogToMapTransformer.TIMESTAMP, SyslogToMapTransformer.HOST,
52  				SyslogToMapTransformer.TAG, SyslogToMapTransformer.MESSAGE };
53  		Object[] values = { 19, 6, date, "WEBERN", "TESTING", "[70729]: TEST SYSLOG MESSAGE" };
54  		assertIterationOrder(fields, values, transformed);
55  	}
56

73  	@Test
74  	public void testWithoutTag() throws Exception {
75  		String syslog = "<158>JUL 26 22:08:35 WEBERN [70729]: TEST SYSLOG MESSAGE";
76  		Map<String, ?> transformed = sut.transformPayload(syslog.getBytes());
77  		assertThat(transformed.size()).isEqualTo(5);
78  		assertThat(transformed.get(SyslogToMapTransformer.FACILITY)).isEqualTo(19);
79  		assertThat(transformed.get(SyslogToMapTransformer.SEVERITY)).isEqualTo(6);
80  		Object date = transformed.get(SyslogToMapTransformer.TIMESTAMP);
81  		assertThat(date instanceof Date || date instanceof String).isTrue();
82  		assertThat(transformed.get(SyslogToMapTransformer.HOST)).isEqualTo("WEBERN");
83  		assertThat(transformed.containsKey(SyslogToMapTransformer.TAG)).isFalse();
84  		assertThat(transformed.get(SyslogToMapTransformer.MESSAGE)).isEqualTo("[70729]: TEST SYSLOG MESSAGE");
85  
86  		String[] fields = { SyslogToMapTransformer.FACILITY, SyslogToMapTransformer.SEVERITY,
87  				SyslogToMapTransformer.TIMESTAMP, SyslogToMapTransformer.HOST,
88  				SyslogToMapTransformer.MESSAGE };
89  		Object[] values = { 19, 6, date, "WEBERN", "[70729]: TEST SYSLOG MESSAGE" };
90  		assertIterationOrder(fields, values, transformed);
91  	}

93  	@Test
94  	public void testTagMaxLength() throws Exception {
95  		String syslog = "<158>JUL 26 22:08:35 WEBERN ABCDE1234567890ABCDE1234567890UVXYZ TEST SYSLOG MESSAGE";
96  		Map<String, ?> transformed = sut.transformPayload(syslog.getBytes());
97  		assertThat(transformed.size()).isEqualTo(6);
98  		assertThat(transformed.get(SyslogToMapTransformer.FACILITY)).isEqualTo(19);
99  		assertThat(transformed.get(SyslogToMapTransformer.SEVERITY)).isEqualTo(6);
100 		Object date = transformed.get(SyslogToMapTransformer.TIMESTAMP);
101 		assertThat(date instanceof Date || date instanceof String).isTrue();
102 		assertThat(transformed.get(SyslogToMapTransformer.HOST)).isEqualTo("WEBERN");
103 		assertThat(transformed.get(SyslogToMapTransformer.TAG)).isEqualTo("ABCDE1234567890ABCDE1234567890UV");
104 		assertThat(transformed.get(SyslogToMapTransformer.MESSAGE)).isEqualTo("XYZ TEST SYSLOG MESSAGE");
105 
106 		String[] fields = { SyslogToMapTransformer.FACILITY, SyslogToMapTransformer.SEVERITY,
107 				SyslogToMapTransformer.TIMESTAMP, SyslogToMapTransformer.HOST,
108 				SyslogToMapTransformer.TAG, SyslogToMapTransformer.MESSAGE };
109 		Object[] values = { 19, 6, date, "WEBERN", "ABCDE1234567890ABCDE1234567890UV", "XYZ TEST SYSLOG MESSAGE" };
110 		assertIterationOrder(fields, values, transformed);
111 	}
112 
113 	private static void assertIterationOrder(String[] expectedFields, Object[] expectedValues,
114 			Map<String, ?> actualTransformed) {
115 		int n = 0;
116 		for (Entry<String, ?> entry : actualTransformed.entrySet()) {
117 			assertThat(entry.getKey()).isEqualTo(expectedFields[n++]);
118 		}
119 		n = 0;
120 		for (String key : actualTransformed.keySet()) {
121 			assertThat(key).isEqualTo(expectedFields[n++]);
122 		}
123 		n = 0;
124 		for (Object value : actualTransformed.values()) {
125 			assertThat(value).isEqualTo(expectedValues[n++]);
126 		}
127 	}
