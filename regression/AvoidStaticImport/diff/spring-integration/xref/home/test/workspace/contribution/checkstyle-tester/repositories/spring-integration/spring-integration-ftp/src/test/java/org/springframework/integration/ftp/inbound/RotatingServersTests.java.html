86  			return null;

96  	@Test
97  	public void testStandard() throws Exception {
98  		AnnotationConfigApplicationContext ctx = new AnnotationConfigApplicationContext(StandardConfig.class);
99  		StandardConfig config = ctx.getBean(StandardConfig.class);
100 		ctx.getBean(StandardIntegrationFlow.class).stop();
101 		assertThat(config.latch.await(10, TimeUnit.SECONDS)).isTrue();
102 		List<Integer> sfCalls = config.sessionSources.stream().limit(17).collect(Collectors.toList());
103 		assertThat(sfCalls).containsExactly(1, 1, 1, 2, 2, 2, 3, 3, 3, 1, 1, 2, 2, 3, 3, 1, 1);
104 		File f1 = new File(tmpDir + File.separator + "standard" + File.separator + "f1");
105 		assertThat(f1.exists()).isTrue();
106 		File f2 = new File(tmpDir + File.separator + "standard" + File.separator + "f2");
107 		assertThat(f2.exists()).isTrue();
108 		File f3 = new File(tmpDir + File.separator + "standard" + File.separator + "f3");
109 		assertThat(f3.exists()).isTrue();
110 		assertThat(ctx.getBean("files", QueueChannel.class).getQueueSize()).isEqualTo(3);
111 		ctx.close();
112 	}

164 	@Configuration
165 	@EnableIntegration
166 	public static class StandardConfig {
167 
168 		private final CountDownLatch latch = new CountDownLatch(17);
169 
170 		final List<Integer> sessionSources = new CopyOnWriteArrayList<>();
171 
172 		@Bean
173 		public SessionFactory<FTPFile> factory1() {
174 			return new CachingSessionFactory<FTPFile>(rawSessionFactory()) {
175 
176 				@Override
177 				public Session<FTPFile> getSession() {
178 					StandardConfig.this.sessionSources.add(1);
179 					StandardConfig.this.latch.countDown();
180 					return super.getSession();
181 				}
182 
183 			};
184 		}
185 
186 		@Bean
187 		public SessionFactory<FTPFile> factory2() {
188 			return new CachingSessionFactory<FTPFile>(rawSessionFactory()) {
189 
190 				@Override
191 				public Session<FTPFile> getSession() {
192 					StandardConfig.this.sessionSources.add(2);
193 					StandardConfig.this.latch.countDown();
194 					return super.getSession();
195 				}
196 
197 			};
198 		}
199 
200 		@Bean
201 		public SessionFactory<FTPFile> factory3() {
202 			return new CachingSessionFactory<FTPFile>(rawSessionFactory()) {
203 
204 				@Override
205 				public Session<FTPFile> getSession() {
206 					StandardConfig.this.sessionSources.add(3);
207 					StandardConfig.this.latch.countDown();
208 					return super.getSession();
209 				}
210 
211 			};
212 		}
213 
214 		@Bean
215 		public SessionFactoryLocator<FTPFile> factoryLocator() {
216 			return new DefaultSessionFactoryLocator<>(new SessionFactoryMapBuilder<FTPFile>()
217 					.put("one", factory1())
218 					.put("two", factory2())
219 					.put("three", factory3())
220 					.get());
221 		}
222 
223 		@Bean
224 		public DelegatingSessionFactory<FTPFile> sf() {
225 			return new DelegatingSessionFactory<>(factoryLocator());
226 		}
227 
228 		@Bean
229 		public RotatingServerAdvice advice() {
230 			List<RotationPolicy.KeyDirectory> keyDirectories = new ArrayList<>();
231 			keyDirectories.add(new RotationPolicy.KeyDirectory("one", "foo"));
232 			keyDirectories.add(new RotationPolicy.KeyDirectory("one", "bar"));
233 			keyDirectories.add(new RotationPolicy.KeyDirectory("two", "baz"));
234 			keyDirectories.add(new RotationPolicy.KeyDirectory("two", "qux"));
235 			keyDirectories.add(new RotationPolicy.KeyDirectory("three", "fiz"));
236 			keyDirectories.add(new RotationPolicy.KeyDirectory("three", "buz"));
237 			return theAdvice(keyDirectories);
238 		}
239 
240 		protected RotatingServerAdvice theAdvice(List<RotationPolicy.KeyDirectory> keyDirectories) {
241 			return new RotatingServerAdvice(sf(), keyDirectories);
242 		}
243 
244 		@Bean
245 		public IntegrationFlow flow() {
246 			return IntegrationFlows.from(Ftp.inboundAdapter(sf())
247 							.filter(new FtpPersistentAcceptOnceFileListFilter(new SimpleMetadataStore(), "rotate"))
248 							.localDirectory(localDir())
249 							.remoteDirectory("."),
250 						e -> e.poller(Pollers.fixedDelay(1).advice(advice())))
251 					.channel(MessageChannels.queue("files"))
252 					.get();
253 		}
254 
255 		protected File localDir() {
256 			return new File(tmpDir, "standard");
257 		}
258 
259 	}

275 	@Configuration
276 	public static class VariableLocalConfig extends StandardConfig {
277 
278 		@Override
279 		@Bean
280 		public IntegrationFlow flow() {
281 			return IntegrationFlows.from(Ftp.inboundAdapter(sf())
282 							.filter(new FtpPersistentAcceptOnceFileListFilter(new SimpleMetadataStore(), "rotate"))
283 							.localDirectory(new File(tmpDir, "variable"))
284 							.localFilenameExpression("#remoteDirectory + T(java.io.File).separator + #root")
285 							.remoteDirectory("."),
286 						e -> e.poller(Pollers.fixedDelay(1).advice(advice())))
287 					.channel(MessageChannels.queue("files"))
288 					.get();
289 		}
