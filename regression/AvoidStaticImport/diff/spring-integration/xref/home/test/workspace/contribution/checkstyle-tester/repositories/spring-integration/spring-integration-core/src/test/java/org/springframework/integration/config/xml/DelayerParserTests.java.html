62  	public void defaultScheduler() {
63  		DelayHandler delayHandler = context.getBean("delayerWithDefaultScheduler.handler", DelayHandler.class);
64  		assertThat(delayHandler.getOrder()).isEqualTo(99);
65  		assertThat(delayHandler.getOutputChannel()).isSameAs(this.context.getBean("output"));
66  		assertThat(TestUtils.getPropertyValue(delayHandler, "defaultDelay", Long.class)).isEqualTo(new Long(1234));
67  		//INT-2243
68  		assertThat(TestUtils.getPropertyValue(delayHandler, "delayExpression")).isNotNull();
69  		assertThat(TestUtils.getPropertyValue(delayHandler, "delayExpression", Expression.class).getExpressionString())
70  				.isEqualTo("headers.foo");
71  		assertThat(TestUtils.getPropertyValue(delayHandler, "messagingTemplate.sendTimeout", Long.class))
72  				.isEqualTo(new Long(987));
73  		assertThat(TestUtils.getPropertyValue(delayHandler, "taskScheduler")).isNull();
74  	}

76  	@Test
77  	public void customScheduler() {
78  		Object endpoint = context.getBean("delayerWithCustomScheduler");
79  		assertThat(endpoint.getClass()).isEqualTo(EventDrivenConsumer.class);
80  		Object handler = TestUtils.getPropertyValue(endpoint, "handler");
81  		assertThat(handler.getClass()).isEqualTo(DelayHandler.class);
82  		DelayHandler delayHandler = (DelayHandler) handler;
83  		assertThat(delayHandler.getOrder()).isEqualTo(Ordered.LOWEST_PRECEDENCE);
84  		DirectFieldAccessor accessor = new DirectFieldAccessor(delayHandler);
85  		assertThat(delayHandler.getOutputChannel()).isSameAs(this.context.getBean("output"));
86  		assertThat(accessor.getPropertyValue("defaultDelay")).isEqualTo(0L);
87  		assertThat(accessor.getPropertyValue("taskScheduler")).isEqualTo(context.getBean("testScheduler"));
88  		assertThat(accessor.getPropertyValue("taskScheduler")).isNotNull();
89  		assertThat(new DirectFieldAccessor(
90  				accessor.getPropertyValue("taskScheduler")).getPropertyValue("waitForTasksToCompleteOnShutdown"))
91  				.isEqualTo(Boolean.TRUE);
92  	}

105 	@Test //INT-2649
106 	public void transactionalSubElement() throws Exception {
107 		Object endpoint = context.getBean("delayerWithTransactional");
108 		DelayHandler delayHandler = TestUtils.getPropertyValue(endpoint, "handler", DelayHandler.class);
109 		List<?> adviceChain = TestUtils.getPropertyValue(delayHandler, "delayedAdviceChain", List.class);
110 		assertThat(adviceChain.size()).isEqualTo(1);
111 		Object advice = adviceChain.get(0);
112 		assertThat(advice instanceof TransactionInterceptor).isTrue();
113 		TransactionAttributeSource transactionAttributeSource = ((TransactionInterceptor) advice).getTransactionAttributeSource();
114 		assertThat(transactionAttributeSource instanceof MatchAlwaysTransactionAttributeSource).isTrue();
115 		Method method = MessageHandler.class.getMethod("handleMessage", Message.class);
116 		TransactionDefinition definition = transactionAttributeSource.getTransactionAttribute(method, null);
117 		assertThat(definition.getPropagationBehavior()).isEqualTo(TransactionDefinition.PROPAGATION_REQUIRED);
118 		assertThat(definition.getIsolationLevel()).isEqualTo(TransactionDefinition.ISOLATION_DEFAULT);
119 		assertThat(definition.getTimeout()).isEqualTo(TransactionDefinition.TIMEOUT_DEFAULT);
120 		assertThat(definition.isReadOnly()).isFalse();
121 	}

123 	@Test //INT-2649
124 	public void adviceChainSubElement() {
125 		Object endpoint = context.getBean("delayerWithAdviceChain");
126 		DelayHandler delayHandler = TestUtils.getPropertyValue(endpoint, "handler", DelayHandler.class);
127 		List<?> adviceChain = TestUtils.getPropertyValue(delayHandler, "delayedAdviceChain", List.class);
128 		assertThat(adviceChain.size()).isEqualTo(2);
129 		assertThat(adviceChain.get(0)).isSameAs(context.getBean("testAdviceBean"));
130 
131 		Object txAdvice = adviceChain.get(1);
132 		assertThat(txAdvice.getClass()).isEqualTo(TransactionInterceptor.class);
133 		TransactionAttributeSource transactionAttributeSource = ((TransactionInterceptor) txAdvice).getTransactionAttributeSource();
134 		assertThat(transactionAttributeSource.getClass()).isEqualTo(NameMatchTransactionAttributeSource.class);
135 		HashMap<?, ?> nameMap = TestUtils.getPropertyValue(transactionAttributeSource, "nameMap", HashMap.class);
136 		assertThat(nameMap.toString()).isEqualTo("{*=PROPAGATION_REQUIRES_NEW,ISOLATION_DEFAULT,readOnly}");
137 	}
