52  @ContextConfiguration
53  @RunWith(SpringJUnit4ClassRunner.class)
54  @DirtiesContext(classMode = ClassMode.AFTER_EACH_TEST_METHOD)
55  public class SplitterIntegrationTests {
56  
57  	@Autowired
58  	MessageChannel inAnnotated;
59  
60  	@Autowired
61  	MessageChannel inMethodInvoking;
62  
63  	@Autowired
64  	MessageChannel inDefault;
65  
66  	@Autowired
67  	MessageChannel inDelimiters;
68  
69  	@Autowired
70  	@Qualifier("splitter.handler")
71  	MethodInvokingSplitter splitter;
72  
73  	private final String sentence = "The quick brown fox jumped over the lazy dog";
74  
75  	private final List<String> words = Arrays.asList(sentence.split("\\s"));
76  
77  	@Autowired
78  	Receiver receiver;
79  
80  	@Before
81  	public void clearWords() {
82  		receiver.receivedWords.clear();
83  	}
84  
85  	@MessageEndpoint
86  	public static class Receiver {
87  		private final List<String> receivedWords = new ArrayList<String>();
88  
89  		@ServiceActivator(inputChannel = "out")
90  		public void deliveredWords(String string) {
91  			this.receivedWords.add(string);
92  		}
93  	}
94  
95  	@MessageEndpoint
96  	public static class TestSplitter {
97  
98  		@Splitter(inputChannel = "inAnnotated", outputChannel = "out")
99  		public Iterator<String> split(String sentence) {
100 			return Arrays.asList(sentence.split("\\s")).iterator();
101 		}
102 	}
103 
104 
105 	@Test
106 	public void configOk() throws Exception {
107 		// just checking the parsing
108 	}
109 
110 	@Test
111 	public void annotated() throws Exception {
112 		inAnnotated.send(new GenericMessage<String>(sentence));
113 		assertThat(this.receiver.receivedWords.containsAll(words)).isTrue();
114 		assertThat(words.containsAll(this.receiver.receivedWords)).isTrue();
115 	}
116 
117 	@Test
118 	public void methodInvoking() throws Exception {
119 		inMethodInvoking.send(new GenericMessage<String>(sentence));
120 		assertThat(receiver.receivedWords.containsAll(words)).isTrue();
121 		assertThat(words.containsAll(this.receiver.receivedWords)).isTrue();
122 	}
123 
124 	@Test
125 	public void defaultSplitter() throws Exception {
126 		inDefault.send(new GenericMessage<List<String>>(words));
127 		assertThat(receiver.receivedWords.containsAll(words)).isTrue();
128 		assertThat(words.containsAll(receiver.receivedWords)).isTrue();
129 	}
130 
131 	@Test
132 	public void delimiterSplitter() throws Exception {
133 		inDelimiters.send(new GenericMessage<String>("one,two, three; four/five"));
134 		assertThat(receiver.receivedWords.containsAll(Arrays.asList("one", "two", "three", "four", "five"))).isTrue();
135 	}
136 
137 	@Test(expected = IllegalArgumentException.class)
138 	public void delimitersNotAllowedWithRef() throws Throwable {
139 		try {
140 			new ClassPathXmlApplicationContext("SplitterIntegrationTests-invalidRef.xml",
141 					SplitterIntegrationTests.class).close();
142 		}
143 		catch (BeanCreationException e) {
144 			Throwable cause = e.getMostSpecificCause();
145 			assertThat(cause).isNotNull();
146 			assertThat(cause instanceof IllegalArgumentException).isTrue();
147 			assertThat(cause.getMessage().contains("'delimiters' property is only available")).isTrue();
148 			throw cause;
149 		}
150 	}
151

153 	public void delimitersNotAllowedWithInnerBean() throws Throwable {

155 			new ClassPathXmlApplicationContext("SplitterIntegrationTests-invalidInnerBean.xml",
156 					SplitterIntegrationTests.class).close();

158 		catch (BeanCreationException e) {
159 			Throwable cause = e.getMostSpecificCause();

162 			assertThat(cause.getMessage().contains("'delimiters' property is only available")).isTrue();
163 			throw cause;
164 		}
165 	}
166

168 	public void delimitersNotAllowedWithExpression() throws Throwable {
169 		try {
170 			new ClassPathXmlApplicationContext("SplitterIntegrationTests-invalidExpression.xml",
171 					SplitterIntegrationTests.class).close();
172 		}

174 			Throwable cause = e.getMostSpecificCause();
175 			assertThat(cause).isNotNull();

177 			assertThat(cause.getMessage().contains("'delimiters' property is only available")).isTrue();
178 			throw cause;
179 		}
180 	}

182 	@Test
183 	public void channelResolver_isNotNull() throws Exception {
184 		splitter.setOutputChannel(null);
185 		Message<String> message = MessageBuilder.withPayload("fooBar")
186 				.setReplyChannelName("out").build();
187 		inMethodInvoking.send(message);
