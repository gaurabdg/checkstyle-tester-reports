93  	public void testObtainConnectionIds(AbstractServerConnectionFactory serverFactory) throws Exception {
94  		final List<IpIntegrationEvent> events =
95  				Collections.synchronizedList(new ArrayList<IpIntegrationEvent>());
96  		int expectedEvents = serverFactory instanceof TcpNetServerConnectionFactory
97  				? 7  // Listening, + OPEN, CLOSE, EXCEPTION for each side
98  				: 5; // Listening, + OPEN, CLOSE (but we *might* get exceptions, depending on timing).
99  		final CountDownLatch serverListeningLatch = new CountDownLatch(1);
100 		final CountDownLatch eventLatch = new CountDownLatch(expectedEvents);
101 		ApplicationEventPublisher publisher = new ApplicationEventPublisher() {
102 
103 			@Override
104 			public void publishEvent(ApplicationEvent event) {
105 				LogFactory.getLog(this.getClass()).trace("Received: " + event);
106 				events.add((IpIntegrationEvent) event);
107 				if (event instanceof TcpConnectionServerListeningEvent) {
108 					serverListeningLatch.countDown();
109 				}
110 				eventLatch.countDown();
111 			}
112 
113 			@Override
114 			public void publishEvent(Object event) {
115 
116 			}
117 
118 		};
119 		serverFactory.setBeanName("serverFactory");
120 		serverFactory.setApplicationEventPublisher(publisher);
121 		serverFactory = spy(serverFactory);
122 		final CountDownLatch serverConnectionInitLatch = new CountDownLatch(1);
123 		doAnswer(invocation -> {
124 			Object result = invocation.callRealMethod();
125 			serverConnectionInitLatch.countDown();
126 			return result;
127 		}).when(serverFactory).wrapConnection(any(TcpConnectionSupport.class));
128 		ThreadPoolTaskScheduler scheduler = new ThreadPoolTaskScheduler();
129 		scheduler.setPoolSize(10);
130 		scheduler.afterPropertiesSet();
131 		BeanFactory bf = mock(BeanFactory.class);
132 		when(bf.containsBean(IntegrationContextUtils.TASK_SCHEDULER_BEAN_NAME)).thenReturn(true);
133 		when(bf.getBean(IntegrationContextUtils.TASK_SCHEDULER_BEAN_NAME, TaskScheduler.class)).thenReturn(scheduler);
134 		serverFactory.setBeanFactory(bf);
135 		TcpReceivingChannelAdapter adapter = new TcpReceivingChannelAdapter();
136 		adapter.setOutputChannel(new NullChannel());
137 		adapter.setConnectionFactory(serverFactory);
138 		adapter.start();
139 		assertThat(serverListeningLatch.await(10, TimeUnit.SECONDS)).as("Listening event not received").isTrue();
140 		assertThat(events.get(0)).isInstanceOf(TcpConnectionServerListeningEvent.class);
141 		assertThat(((TcpConnectionServerListeningEvent) events.get(0)).getPort()).isEqualTo(serverFactory.getPort());
142 		int port = serverFactory.getPort();
143 		TcpNetClientConnectionFactory clientFactory = new TcpNetClientConnectionFactory("localhost", port);
144 		clientFactory.registerListener(message -> false);
145 		clientFactory.setBeanName("clientFactory");
146 		clientFactory.setApplicationEventPublisher(publisher);
147 		clientFactory.start();
148 		TcpConnectionSupport client = clientFactory.getConnection();
149 		List<String> clients = clientFactory.getOpenConnectionIds();
150 		assertThat(clients.size()).isEqualTo(1);
151 		assertThat(clients.contains(client.getConnectionId())).isTrue();
152 		assertThat(serverConnectionInitLatch.await(10, TimeUnit.SECONDS)).as("Server connection failed to register")
153 				.isTrue();
154 		List<String> servers = serverFactory.getOpenConnectionIds();
155 		assertThat(servers.size()).isEqualTo(1);
156 		assertThat(serverFactory.closeConnection(servers.get(0))).isTrue();
157 		servers = serverFactory.getOpenConnectionIds();
158 		assertThat(servers.size()).isEqualTo(0);
159 		int n = 0;
160 		clients = clientFactory.getOpenConnectionIds();
161 		while (n++ < 100 && clients.size() > 0) {
162 			Thread.sleep(100);
163 			clients = clientFactory.getOpenConnectionIds();
164 		}
165 		assertThat(clients.size()).isEqualTo(0);
166 		assertThat(eventLatch.await(10, TimeUnit.SECONDS)).isTrue();
167 		assertThat(events.size())
168 				.as("Expected at least " + expectedEvents + " events; got: " + events.size() + " : " + events)
169 				.isGreaterThanOrEqualTo(expectedEvents);
170 
171 		FooEvent event = new FooEvent(client, "foo");
172 		client.publishEvent(event);
173 		assertThat(events.size())
174 				.as("Expected at least " + expectedEvents + " events; got: " + events.size() + " : " + events)
175 				.isGreaterThanOrEqualTo(expectedEvents + 1);
176 
177 		try {
178 			event = new FooEvent(mock(TcpConnectionSupport.class), "foo");
179 			client.publishEvent(event);
180 			fail("Expected exception");
181 		}
182 		catch (IllegalArgumentException e) {
183 			assertThat("Can only publish events with this as the source".equals(e.getMessage())).isTrue();
184 		}
185 
186 		SocketAddress address = serverFactory.getServerSocketAddress();
187 		if (address instanceof InetSocketAddress) {
188 			InetSocketAddress inetAddress = (InetSocketAddress) address;
189 			assertThat(inetAddress.getPort()).isEqualTo(port);
190 		}
191 		serverFactory.stop();
192 		scheduler.shutdown();
193 	}

207 	private void testEarlyClose(final AbstractServerConnectionFactory factory, String property,
208 			String message) throws Exception {
209 		factory.setApplicationEventPublisher(mock(ApplicationEventPublisher.class));
210 		factory.setBeanName("foo");
211 		factory.registerListener(mock(TcpListener.class));
212 		factory.afterPropertiesSet();
213 		Log logger = spy(TestUtils.getPropertyValue(factory, "logger", Log.class));
214 		new DirectFieldAccessor(factory).setPropertyValue("logger", logger);
215 		final CountDownLatch latch1 = new CountDownLatch(1);
216 		final CountDownLatch latch2 = new CountDownLatch(1);
217 		final CountDownLatch latch3 = new CountDownLatch(1);
218 		when(logger.isInfoEnabled()).thenReturn(true);
219 		when(logger.isDebugEnabled()).thenReturn(true);
220 		doAnswer(invocation -> {
221 			latch1.countDown();
222 			// wait until the stop nulls the channel
223 			latch2.await(10, TimeUnit.SECONDS);
224 			return null;
225 		}).when(logger).info(contains("Listening"));
226 		doAnswer(invocation -> {
227 			latch3.countDown();
228 			return null;
229 		}).when(logger).debug(contains(message));
230 		factory.start();
231 		assertThat(latch1.await(10, TimeUnit.SECONDS)).as("missing info log").isTrue();
232 		// stop on a different thread because it waits for the executor
233 		new SimpleAsyncTaskExecutor()
234 				.execute(factory::stop);
235 		int n = 0;
236 		DirectFieldAccessor accessor = new DirectFieldAccessor(factory);
237 		while (n++ < 200 && accessor.getPropertyValue(property) != null) {
238 			Thread.sleep(100);
239 		}
240 		assertThat(n < 200).as("Stop was not invoked in time").isTrue();
241 		latch2.countDown();
242 		assertThat(latch3.await(10, TimeUnit.SECONDS)).as("missing debug log").isTrue();
243 		String expected = "bean 'foo', port=" + factory.getPort() + message;
244 		ArgumentCaptor<String> captor = ArgumentCaptor.forClass(String.class);
245 		verify(logger, atLeast(1)).debug(captor.capture());
246 		assertThat(captor.getAllValues()).contains(expected);
247 		factory.stop();
248 	}
