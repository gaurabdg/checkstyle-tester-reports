40  	@Test
41  	public void testStringSerialization() throws IOException {
42  		String str = "hello";
43  		PojoCodec codec = new PojoCodec();
44  		ByteArrayOutputStream bos = new ByteArrayOutputStream();
45  
46  		codec.encode(str, bos);
47  
48  		String s2 = codec.decode(bos.toByteArray(), String.class);
49  		assertThat(s2).isEqualTo(str);
50  	}
51  
52  	@Test
53  	public void testSerializationWithStreams() throws IOException {
54  		String str = "hello";
55  		File file = new File("test.ser");
56  		PojoCodec codec = new PojoCodec();
57  		FileOutputStream fos = new FileOutputStream(file);
58  		codec.encode(str, fos);
59  		fos.close();
60  
61  		FileInputStream fis = new FileInputStream(file);
62  		String s2 = codec.decode(fis, String.class);
63  		file.delete();
64  		assertThat(s2).isEqualTo(str);
65  	}

67  	@Test
68  	public void testPojoSerialization() throws IOException {
69  		PojoCodec codec = new PojoCodec();
70  		SomeClassWithNoDefaultConstructors foo = new SomeClassWithNoDefaultConstructors("foo", 123);
71  		ByteArrayOutputStream bos = new ByteArrayOutputStream();
72  		codec.encode(foo, bos);
73  		Object foo2 = codec.decode(bos.toByteArray(), SomeClassWithNoDefaultConstructors.class);
74  		assertThat(foo2).isEqualTo(foo);
75  	}

77  	@Test
78  	public void testPrimitiveSerialization() throws IOException {
79  		PojoCodec codec = new PojoCodec();
80  
81  		ByteArrayOutputStream bos = new ByteArrayOutputStream();
82  		codec.encode(true, bos);
83  		boolean b = codec.decode(bos.toByteArray(), Boolean.class);
84  		assertThat(b).isEqualTo(true);
85  		b = codec.decode(bos.toByteArray(), boolean.class);
86  		assertThat(b).isEqualTo(true);
87  
88  		bos = new ByteArrayOutputStream();
89  		codec.encode(3.14159, bos);
90  
91  		double d = codec.decode(bos.toByteArray(), double.class);
92  		assertThat(d).isCloseTo(3.14159, Offset.offset(0.00001));
93  
94  		bos = new ByteArrayOutputStream();
95  		codec.encode(3.14159, bos);
96  
97  		d = codec.decode(bos.toByteArray(), Double.class);
98  		assertThat(d).isCloseTo(3.14159, Offset.offset(0.00001));
99  
100 	}

102 	@Test
103 	public void testMapSerialization() throws IOException {
104 		PojoCodec codec = new PojoCodec();
105 		Map<String, Integer> map = new HashMap<>();
106 		map.put("one", 1);
107 		map.put("two", 2);
108 		ByteArrayOutputStream bos = new ByteArrayOutputStream();
109 		codec.encode(map, bos);
110 		Map<?, ?> m2 = (Map<?, ?>) codec.decode(bos.toByteArray(), HashMap.class);
111 		assertThat(m2.size()).isEqualTo(2);
112 		assertThat(m2.get("one")).isEqualTo(1);
113 		assertThat(m2.get("two")).isEqualTo(2);
114 	}

116 	@Test
117 	public void testComplexObjectSerialization() throws IOException {
118 		PojoCodec codec = new PojoCodec();
119 		Foo foo = new Foo();
120 		foo.put("one", 1);
121 		foo.put("two", 2);
122 		ByteArrayOutputStream bos = new ByteArrayOutputStream();
123 		codec.encode(foo, bos);
124 
125 		Foo foo2 = codec.decode(bos.toByteArray(), Foo.class);
126 		assertThat(foo2.get("one")).isEqualTo(1);
127 		assertThat(foo2.get("two")).isEqualTo(2);
128 	}

130 	static class SomeClassWithNoDefaultConstructors {
131 
132 		private String val1;
133 
134 		private int val2;
135 
136 		SomeClassWithNoDefaultConstructors(String val1, int val2) {
137 			this.val1 = val1;
138 			this.val2 = val2;
139 		}
140 
141 		@Override
142 		public boolean equals(Object other) {
143 			if (!(other instanceof SomeClassWithNoDefaultConstructors)) {
144 				return false;
145 			}
146 			SomeClassWithNoDefaultConstructors that = (SomeClassWithNoDefaultConstructors) other;
147 			return (this.val1.equals(that.val1) && this.val2 == that.val2);
148 		}
149 
150 		@Override
151 		public int hashCode() {
152 			int result = this.val1.hashCode();
153 			result = 31 * result + this.val2;
154 			return result;
155 		}
156 
157 	}

168 			this.map.put(key, value);
