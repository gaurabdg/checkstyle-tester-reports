57  @RunWith(SpringJUnit4ClassRunner.class)
58  @ContextConfiguration
59  public class MarshallingWebServiceIntegrationTests {
60  
61  	private static final String input = "<hello/>";
62  
63  	@Autowired
64  	MarshallingWebServiceInboundGateway gateway;
65  
66  	@Mock
67  	private MessageContext context;
68  
69  	@Mock
70  	private WebServiceMessage response;
71  
72  	@Mock
73  	private WebServiceMessage request;
74  
75  	private Source stringSource = new StreamSource(new StringReader(input));
76  
77  	private StringWriter output = new StringWriter();
78  
79  	private Result stringResult = new StreamResult(output);
80  
81  	@Before
82  	public void setupMocks() {
83  		MockitoAnnotations.initMocks(this);
84  	}
85  
86  	@Test
87  	public void configOk() throws Exception {
88  		// just flag invalid config
89  	}
90  
91  	@Test
92  	public void sendString() throws Exception {
93  		when(context.getResponse()).thenReturn(response);
94  		when(context.getRequest()).thenReturn(request);
95  		when(request.getPayloadSource()).thenReturn(stringSource);
96  		when(response.getPayloadResult()).thenReturn(stringResult);
97  		gateway.invoke(context);
98  		assertThat(output.toString().endsWith(input)).isTrue();
99  	}
100 
101 	public static class StubMarshaller implements Marshaller, Unmarshaller {
102 
103 		public void marshal(Object graph, Result result)
104 				throws XmlMappingException, IOException {
105 			Transformer transformer;
106 			try {
107 				transformer = TransformerFactory.newInstance().newTransformer();
108 				StringSource stringSource = new StringSource(graph.toString());
109 				transformer.transform(stringSource, result);
110 			}
111 			catch (Exception e) {
112 				throw new MarshallingFailureException("Stub failed to marshal",
113 						e);
114 			}
115 		}
116 
117 
118 		@SuppressWarnings("rawtypes")
119 		public boolean supports(Class clazz) {
120 			return true;
121 		}
122 
123 		public Object unmarshal(Source source) throws XmlMappingException,
124 				IOException {
125 			//this is a hack, but we're not here to test marshalling
126 			return input;
127 		}
128 	}
