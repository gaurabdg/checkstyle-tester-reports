116 	@Test
117 	public void withHeaderMapperCustomHeaders() {
118 		Object eventDrivenConsumer = context.getBean("withHeaderMapperCustomHeaders");
119 
120 		AmqpOutboundEndpoint endpoint = TestUtils.getPropertyValue(eventDrivenConsumer, "handler",
121 				AmqpOutboundEndpoint.class);
122 		assertThat(TestUtils.getPropertyValue(endpoint, "defaultDeliveryMode")).isNotNull();
123 		assertThat(TestUtils.getPropertyValue(endpoint, "lazyConnect", Boolean.class)).isFalse();
124 		assertThat(TestUtils
125 				.getPropertyValue(endpoint, "delayExpression", org.springframework.expression.Expression.class)
126 				.getExpressionString()).isEqualTo("42");
127 		assertThat(TestUtils.getPropertyValue(endpoint, "headersMappedLast", Boolean.class)).isFalse();
128 
129 		Field amqpTemplateField = ReflectionUtils.findField(AmqpOutboundEndpoint.class, "rabbitTemplate");
130 		amqpTemplateField.setAccessible(true);
131 		RabbitTemplate amqpTemplate = TestUtils.getPropertyValue(endpoint, "rabbitTemplate", RabbitTemplate.class);
132 		amqpTemplate = Mockito.spy(amqpTemplate);
133 		final AtomicBoolean shouldBePersistent = new AtomicBoolean();
134 
135 		Mockito.doAnswer(invocation -> {
136 			Object[] args = invocation.getArguments();
137 			org.springframework.amqp.core.Message amqpMessage = (org.springframework.amqp.core.Message) args[2];
138 			MessageProperties properties = amqpMessage.getMessageProperties();
139 			assertThat(properties.getHeaders().get("foo")).isEqualTo("foo");
140 			assertThat(properties.getHeaders().get("foobar")).isEqualTo("foobar");
141 			assertThat(properties.getHeaders().get("bar")).isNull();
142 			assertThat(properties.getDeliveryMode()).isEqualTo(shouldBePersistent.get() ?
143 					MessageDeliveryMode.PERSISTENT
144 					: MessageDeliveryMode.NON_PERSISTENT);
145 			return null;
146 		})
147 				.when(amqpTemplate).send(Mockito.any(String.class), Mockito.any(String.class),
148 				Mockito.any(org.springframework.amqp.core.Message.class), Mockito.any(CorrelationData.class));
149 		ReflectionUtils.setField(amqpTemplateField, endpoint, amqpTemplate);
150 
151 		MessageChannel requestChannel = context.getBean("requestChannel", MessageChannel.class);
152 		Message<?> message = MessageBuilder.withPayload("hello")
153 				.setHeader("foo", "foo")
154 				.setHeader("bar", "bar")
155 				.setHeader("foobar", "foobar")
156 				.build();
157 		requestChannel.send(message);
158 		Mockito.verify(amqpTemplate, Mockito.times(1)).send(anyString(),
159 				isNull(), Mockito.any(org.springframework.amqp.core.Message.class), isNull());
160 
161 		shouldBePersistent.set(true);
162 		message = MessageBuilder.withPayload("hello")
163 				.setHeader("foo", "foo")
164 				.setHeader("bar", "bar")
165 				.setHeader("foobar", "foobar")
166 				.setHeader(AmqpHeaders.DELIVERY_MODE, MessageDeliveryMode.PERSISTENT)
167 				.build();
168 		requestChannel.send(message);
169 	}
