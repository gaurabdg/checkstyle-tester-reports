82  		DataSource datasource = mock(DataSource.class);

85  		final ExpressionFactoryBean efb = new ExpressionFactoryBean("headers['stored_procedure_name']");

236 	@Test
237 	public void testStoredProcExecutorWithNonResolvingExpression() throws Exception {
238 		final DataSource datasource = mock(DataSource.class);
239 
240 		final StoredProcExecutor storedProcExecutor = new StoredProcExecutor(datasource);
241 
242 		final ExpressionFactoryBean efb = new ExpressionFactoryBean("headers['stored_procedure_name']");
243 		efb.afterPropertiesSet();
244 		final Expression expression = efb.getObject();
245 
246 		storedProcExecutor.setStoredProcedureNameExpression(expression);
247 		storedProcExecutor.setBeanFactory(mock(BeanFactory.class));
248 
249 		storedProcExecutor.afterPropertiesSet();
250 
251 		Map<String, SimpleJdbcCallOperations> jdbcCallOperationsMap = new HashMap<>();
252 		jdbcCallOperationsMap.put("123", mock(SimpleJdbcCallOperations.class));
253 
254 		new DirectFieldAccessor(storedProcExecutor)
255 				.setPropertyValue("jdbcCallOperationsMap", jdbcCallOperationsMap);
256 		//This should work
257 
258 		storedProcExecutor.executeStoredProcedure(
259 				MessageBuilder.withPayload("test")
260 						.setHeader("stored_procedure_name", "123")
261 						.build());
262 
263 		//This should cause an exception
264 
265 		assertThatIllegalArgumentException()
266 				.isThrownBy(() ->
267 						storedProcExecutor.executeStoredProcedure(
268 								MessageBuilder.withPayload("test")
269 										.setHeader("some_other_header", "123")
270 										.build()))
271 				.withMessage("Unable to resolve Stored Procedure/Function name for the provided Expression " +
272 						"'headers['stored_procedure_name']'.");
273 	}
