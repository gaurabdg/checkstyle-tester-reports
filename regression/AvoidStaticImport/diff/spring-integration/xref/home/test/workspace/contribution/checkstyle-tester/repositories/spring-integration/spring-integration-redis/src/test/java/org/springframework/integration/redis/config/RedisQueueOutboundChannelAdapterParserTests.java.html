32  import org.springframework.integration.handler.advice.RequestHandlerRetryAdvice;

34  import org.springframework.integration.test.util.TestUtils;
35  import org.springframework.test.annotation.DirtiesContext;
36  import org.springframework.test.context.ContextConfiguration;
37  import org.springframework.test.context.junit4.SpringJUnit4ClassRunner;
38  
39  /**
40   * @author Artem Bilan
41   * @author Gary Russell
42   * @author Rainer Frey
43   *
44   * @since 3.0
45   */
46  @ContextConfiguration
47  @RunWith(SpringJUnit4ClassRunner.class)
48  @DirtiesContext
49  public class RedisQueueOutboundChannelAdapterParserTests {
50  
51  	@Autowired
52  	@Qualifier("redisConnectionFactory")
53  	private RedisConnectionFactory connectionFactory;
54  
55  	@Autowired
56  	@Qualifier("customRedisConnectionFactory")
57  	private RedisConnectionFactory customRedisConnectionFactory;
58  
59  	@Autowired
60  	@Qualifier("defaultAdapter")
61  	private EventDrivenConsumer defaultEndpoint;
62  
63  	@Autowired
64  	@Qualifier("defaultAdapter.handler")
65  	private RedisQueueOutboundChannelAdapter defaultAdapter;
66  
67  	@Autowired
68  	@Qualifier("customAdapter.handler")
69  	private RedisQueueOutboundChannelAdapter customAdapter;
70  
71  	@Autowired
72  	private RedisSerializer<?> serializer;
73  
74  	@Test
75  	public void testInt3017DefaultConfig() throws Exception {
76  		assertThat(TestUtils.getPropertyValue(this.defaultAdapter, "template.connectionFactory"))
77  				.isSameAs(this.connectionFactory);
78  		assertThat(TestUtils.getPropertyValue(this.defaultAdapter, "queueNameExpression", Expression.class)
79  				.getExpressionString()).isEqualTo("foo");
80  		assertThat(TestUtils.getPropertyValue(this.defaultAdapter, "extractPayload", Boolean.class)).isTrue();
81  		assertThat(TestUtils.getPropertyValue(this.defaultAdapter, "serializerExplicitlySet", Boolean.class)).isFalse();
82  
83  		Object handler = TestUtils.getPropertyValue(this.defaultEndpoint, "handler");
84  
85  		assertThat(AopUtils.isAopProxy(handler)).isTrue();
86  
87  		assertThat(this.defaultAdapter).isSameAs(((Advised) handler).getTargetSource().getTarget());
88  
89  		assertThat(TestUtils.getPropertyValue(handler, "h.advised.advisors[0].advice"))
90  				.isInstanceOf(RequestHandlerRetryAdvice.class);
91  		assertThat(TestUtils.getPropertyValue(this.defaultAdapter, "leftPush", Boolean.class)).isTrue();
92  	}
93  
94  	@Test
95  	public void testInt3017CustomConfig() {
96  		assertThat(TestUtils.getPropertyValue(this.customAdapter, "template.connectionFactory"))
97  				.isSameAs(this.customRedisConnectionFactory);
98  		assertThat(TestUtils.getPropertyValue(this.customAdapter, "queueNameExpression.expression"))
99  				.isEqualTo("headers['redis_queue']");
100 		assertThat(TestUtils.getPropertyValue(this.customAdapter, "extractPayload", Boolean.class)).isFalse();
101 		assertThat(TestUtils.getPropertyValue(this.customAdapter, "serializerExplicitlySet", Boolean.class)).isTrue();
102 		assertThat(TestUtils.getPropertyValue(this.customAdapter, "serializer")).isSameAs(this.serializer);
103 		assertThat(TestUtils.getPropertyValue(this.customAdapter, "leftPush", Boolean.class)).isFalse();
104 	}
105 
106 }
