31  /**
32   * Unit test for {@link AttributeAxisIterator}
33   */
34  public class AttributeAxisIteratorTest {
35  
36      @Rule
37      public JavaUtilLoggingRule loggingRule = new JavaUtilLoggingRule(Attribute.class.getName());
38  
39      /**
40       * Verifies that attributes are returned, even if they are deprecated.
41       * Deprecated attributes are still accessible, but a warning is logged, when
42       * the value is used.
43       */
44      @Test
45      public void testAttributeDeprecation() {
46          // make sure, we log
47          Attribute.DETECTED_DEPRECATED_ATTRIBUTES.clear();
48  
49          Node dummy = new DummyNodeWithDeprecatedAttribute(2);
50          Map<String, Attribute> attributes = toMap(new AttributeAxisIterator(dummy));
51          assertThat(attributes, IsMapContaining.hasKey("Size"));
52          assertThat(attributes, IsMapContaining.hasKey("Name"));
53  
54          assertThat(attributes.get("Size").getStringValue(), Matchers.is("2"));
55          assertThat(attributes.get("Name").getStringValue(), Matchers.is("foo"));
56  
57          String log = loggingRule.getLog();
58          assertThat(log, Matchers.containsString("Use of deprecated attribute 'dummyNode/@Size' in XPath query"));
59          assertThat(log, Matchers.containsString("Use of deprecated attribute 'dummyNode/@Name' in XPath query"));
60      }
61  
62      /**
63       * Test hasNext and next.
64       */
65      @Test
66      public void testAttributeAxisIterator() {
67          DummyNode dummyNode = new DummyNode(1);
68          dummyNode.testingOnlySetBeginLine(1);
69          dummyNode.testingOnlySetBeginColumn(1);
70  
71          AttributeAxisIterator it = new AttributeAxisIterator(dummyNode);
72          Map<String, Attribute> atts = toMap(it);
73          Assert.assertEquals(7, atts.size());
74          assertTrue(atts.containsKey("BeginColumn"));
75          assertTrue(atts.containsKey("BeginLine"));
76          assertTrue(atts.containsKey("FindBoundary"));
77          assertTrue(atts.containsKey("Image"));
78          assertTrue(atts.containsKey("SingleLine"));
79          assertTrue(atts.containsKey("EndColumn"));
80          assertTrue(atts.containsKey("EndLine"));
81      }
82  
83      @Test
84      public void testAttributeAxisIteratorWithEnum() {
85          DummyNodeWithEnum dummyNode = new DummyNodeWithEnum(1);
86  
87          AttributeAxisIterator it = new AttributeAxisIterator(dummyNode);
88          Map<String, Attribute> atts = toMap(it);
89          Assert.assertEquals(8, atts.size());
90          assertTrue(atts.containsKey("Enum"));
91          assertEquals(DummyNodeWithEnum.MyEnum.FOO, atts.get("Enum").getValue());
92      }
93  
94      @Test
95      public void testAttributeAxisIteratorWithList() {
96          DummyNodeWithList dummyNode = new DummyNodeWithList(1);
97  
98          AttributeAxisIterator it = new AttributeAxisIterator(dummyNode);
99          Map<String, Attribute> atts = toMap(it);
100         Assert.assertEquals(8, atts.size());
101         assertTrue(atts.containsKey("List"));
102         assertEquals(Arrays.asList("A", "B"), atts.get("List").getValue());
103         assertFalse(atts.containsKey("NodeList"));
104     }
105 
106     private Map<String, Attribute> toMap(AttributeAxisIterator it) {
107         Map<String, Attribute> atts = new HashMap<>();
108         while (it.hasNext()) {
109             Attribute attribute = it.next();
110             atts.put(attribute.getName(), attribute);
111         }
112         return atts;
113     }
114 
115     public static class DummyNodeWithEnum extends DummyNode {
116 
117         public DummyNodeWithEnum(int id) {
118             super(id);
119         }
120 
121         public enum MyEnum {
122             FOO, BAR
123         }
124 
125         public MyEnum getEnum() {
126             return MyEnum.FOO;
127         }
128     }
129 
130     public static class DummyNodeWithList extends DummyNode {
131 
132         public DummyNodeWithList(int id) {
133             super(id);
134         }
135 
136         public List<String> getList() {
137             return Arrays.asList("A", "B");
138         }
139 
140         public List<Node> getNodeList() {
141             return Collections.emptyList();
142         }
143     }
