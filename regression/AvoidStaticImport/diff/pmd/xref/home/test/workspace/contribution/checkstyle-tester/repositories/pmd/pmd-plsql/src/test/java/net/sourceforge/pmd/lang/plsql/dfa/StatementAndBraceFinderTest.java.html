35      /**
36       * Java ASTStatementExpression equivalent is inferred as an Expression()
37       * which has an UnlabelledStatement as a parent.
38       */
39      @Test
40      public void testExpressionParentChildLinks() {
41          ASTExpression ex = getExpr(TEST1);
42          DataFlowNode dfn = ex.getDataFlowNode();
43          assertEquals(3, dfn.getLine());
44          assertTrue(dfn.getNode() instanceof ASTExpression);
45          List<DataFlowNode> dfns = dfn.getParents();
46          assertEquals(1, dfns.size());
47          DataFlowNode parentDfn = dfns.get(0);
48          assertEquals(2, parentDfn.getLine());
49          assertTrue(parentDfn.getNode() instanceof ASTProgramUnit);
50          ASTProgramUnit exParent = (ASTProgramUnit) parentDfn.getNode();
51          // Validate the two-way link between Program Unit and Statement
52          assertEquals(ex, exParent.getDataFlowNode().getChildren().get(0).getNode());
53          assertEquals(exParent, ex.getDataFlowNode().getParents().get(0).getNode());
54      }

107     @Test
108     public void testSimpleCaseStmtHasCorrectTypes() {
109         ASTExpression exp = getExpr(TEST6);
110         DataFlowNode dfn = null;
111         dfn = exp.getDataFlowNode().getFlow().get(0);
112         assertTrue(dfn instanceof StartOrEndDataFlowNode);
113         dfn = exp.getDataFlowNode().getFlow().get(1);
114         assertEquals(2, dfn.getLine());
115         assertTrue(dfn.getNode() instanceof ASTProgramUnit);
116         dfn = exp.getDataFlowNode().getFlow().get(2);
117         assertEquals(4, dfn.getLine());
118         assertTrue(dfn.isType(NodeType.SWITCH_START));
119         assertTrue(dfn.isType(NodeType.CASE_LAST_STATEMENT));
120         dfn = exp.getDataFlowNode().getFlow().get(3);
121         assertEquals(5, dfn.getLine());
122         assertTrue(dfn.isType(NodeType.CASE_LAST_STATEMENT));
123         assertTrue(dfn.isType(NodeType.BREAK_STATEMENT));
124         dfn = exp.getDataFlowNode().getFlow().get(4);
125         assertEquals(6, dfn.getLine());
126         assertTrue(dfn.isType(NodeType.SWITCH_LAST_DEFAULT_STATEMENT));
127         assertTrue(dfn.isType(NodeType.BREAK_STATEMENT));
128         dfn = exp.getDataFlowNode().getFlow().get(5);
129         assertEquals(7, dfn.getLine());
130         assertTrue(dfn.isType(NodeType.SWITCH_END));
131     }
