7   import static org.junit.Assert.assertEquals;

9   import java.util.Arrays;
10  import java.util.List;

12  import org.junit.Test;

14  public class EscapeUtilsTest {
15  
16      @Test
17      public void testEscapeMarkdown() {
18          assertEquals("This is a \\\\backslash", EscapeUtils.escapeMarkdown("This is a \\backslash"));
19          assertEquals("This \"\\*\" is not a emphasis", EscapeUtils.escapeMarkdown("This \"*\" is not a emphasis"));
20          assertEquals("This \"\\*\\*\" is not a strong style", EscapeUtils.escapeMarkdown("This \"**\" is not a strong style"));
21          assertEquals("This \"\\[foo\\]\" does not start a link", EscapeUtils.escapeMarkdown("This \"[foo]\" does not start a link"));
22          assertEquals("This \"\\~bar\\~\" is not a strike-through", EscapeUtils.escapeMarkdown("This \"~bar~\" is not a strike-through"));
23          assertEquals("That's \"\\|\" just a bar", EscapeUtils.escapeMarkdown("That's \"|\" just a bar"));
24          assertEquals("This \"\\_\" is just a underscore", EscapeUtils.escapeMarkdown("This \"_\" is just a underscore"));
25      }
26  
27      @Test
28      public void testEscapeHtmlWithinMarkdownSingleLine() {
29          assertEquals("a &lt;script&gt; tag outside of `<script>` backticks should be escaped",
30                  EscapeUtils.escapeSingleLine("a <script> tag outside of `<script>` backticks should be escaped"));
31          assertEquals("a &lt;script&gt; &quot;tag&quot; outside of `<script>` backticks should be escaped &lt;multiple&gt; times `<strong>`.",
32                  EscapeUtils.escapeSingleLine("a <script> \"tag\" outside of `<script>` backticks should be escaped <multiple> times `<strong>`."));
33          assertEquals("URLS: a https://pmd.github.io or a <https://pmd.github.io> are turned into links",
34                  EscapeUtils.escapeSingleLine("URLS: a https://pmd.github.io or a <https://pmd.github.io> are turned into links"));
35          assertEquals("multiple URLS: <https://pmd.github.io> and <https://pmd.github.io> are two links",
36                  EscapeUtils.escapeSingleLine("multiple URLS: <https://pmd.github.io> and <https://pmd.github.io> are two links"));
37          assertEquals("URL: <http://www.google.com> is a url without ssl",
38                  EscapeUtils.escapeSingleLine("URL: <http://www.google.com> is a url without ssl"));
39          assertEquals("> this is a quote line",
40                  EscapeUtils.escapeSingleLine("> this is a quote line"));
41          assertEquals("combination of URLs and backticks: <https://pmd.github.io> but `<script>` &lt;strong&gt;escaped&lt;/strong&gt;",
42                  EscapeUtils.escapeSingleLine("combination of URLs and backticks: <https://pmd.github.io> but `<script>` <strong>escaped</strong>"));
43          assertEquals("combination of URLs and backticks: `<script>` &lt;strong&gt;escaped&lt;/strong&gt; but <https://pmd.github.io>",
44                  EscapeUtils.escapeSingleLine("combination of URLs and backticks: `<script>` <strong>escaped</strong> but <https://pmd.github.io>"));
45      }
46  
47      @Test
48      public void testEscapeHtmlWithinMarkdownBlocks() {
49          String text = "paragraph\n\n> quote <script>\n> quote line \"2\"\n>quote line `<script>` 3\n\n"
50                  + "next paragraph\n\n    code <script> \"a < b\"\n    code line 2\n\n"
51                  + "next paragraph\n\n```\ncode <script> \"a < b\"\ncode line 2\n```\n\n"
52                  + "next paragraph\n\n```java\nString = \"code <script> with syntax highlighting\";\ncode line 2\n```\n";
53          String expected = "paragraph\n\n> quote &lt;script&gt;\n> quote line &quot;2&quot;\n>quote line `<script>` 3\n\n"
54                  + "next paragraph\n\n    code <script> \"a < b\"\n    code line 2\n\n"
55                  + "next paragraph\n\n```\ncode <script> \"a < b\"\ncode line 2\n```\n\n"
56                  + "next paragraph\n\n```java\nString = \"code <script> with syntax highlighting\";\ncode line 2\n```\n";
57          List<String> escaped = EscapeUtils.escapeLines(Arrays.asList(text.split("\n")));
58          assertEquals(Arrays.asList(expected.split("\n")), escaped);
59      }
