5   package net.sourceforge.pmd.lang.java.rule;
6   
7   import static net.sourceforge.pmd.properties.PropertyFactory.stringListProperty;
8   
9   import java.util.Collection;
10  import java.util.Collections;
11  import java.util.List;
12  
13  import net.sourceforge.pmd.lang.java.ast.Annotatable;
14  import net.sourceforge.pmd.properties.PropertyDescriptor;
15  
16  public abstract class AbstractIgnoredAnnotationRule extends AbstractJavaRule {
17  
18      private final PropertyDescriptor<List<String>> ignoredAnnotationsDescriptor
19          = stringListProperty("ignoredAnnotations")
20          .desc("Fully qualified names of the annotation types that should be ignored by this rule")
21          .defaultValue(defaultSuppressionAnnotations())
22          .build();
23  
24      protected Collection<String> defaultSuppressionAnnotations() {
25          return Collections.emptyList();
26      }
27  
28      protected AbstractIgnoredAnnotationRule() {
29          definePropertyDescriptor(ignoredAnnotationsDescriptor);
30      }
31  
32  
33      /**
34       * Checks whether any annotation in ignoredAnnotationsDescriptor is present on the node.
35       *
36       * @param node
37       *            the node to check
38       * @return <code>true</code> if the annotation has been found, otherwise <code>false</code>
39       */
40      protected boolean hasIgnoredAnnotation(Annotatable node) {
41          return node.isAnyAnnotationPresent(getProperty(ignoredAnnotationsDescriptor));
42      }
43  }
