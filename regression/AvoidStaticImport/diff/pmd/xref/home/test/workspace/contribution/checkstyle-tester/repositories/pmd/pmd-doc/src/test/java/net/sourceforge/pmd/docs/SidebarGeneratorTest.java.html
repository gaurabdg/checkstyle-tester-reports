22  import org.yaml.snakeyaml.DumperOptions;
23  import org.yaml.snakeyaml.DumperOptions.FlowStyle;
24  import org.yaml.snakeyaml.DumperOptions.LineBreak;
25  import org.yaml.snakeyaml.Yaml;
26  
27  import net.sourceforge.pmd.RuleSet;
28  import net.sourceforge.pmd.RulesetsFactoryUtils;
29  import net.sourceforge.pmd.lang.Language;
30  import net.sourceforge.pmd.lang.LanguageRegistry;
31  
32  public class SidebarGeneratorTest {
33      private MockedFileWriter writer = new MockedFileWriter();
34  
35      @Before
36      public void setup() {
37          writer.reset();
38      }
39  
40      @Test
41      public void testSidebar() throws IOException {
42          Map<Language, List<RuleSet>> rulesets = new HashMap<>();
43          RuleSet ruleSet1 = RulesetsFactoryUtils.defaultFactory().createNewRuleSet("test", "test", "bestpractices.xml", Collections.emptyList(), Collections.emptyList(), Collections.emptyList());
44          RuleSet ruleSet2 = RulesetsFactoryUtils.defaultFactory().createNewRuleSet("test2", "test", "codestyle.xml", Collections.emptyList(), Collections.emptyList(), Collections.emptyList());
45          rulesets.put(LanguageRegistry.findLanguageByTerseName("java"), Arrays.asList(ruleSet1, ruleSet2));
46          rulesets.put(LanguageRegistry.findLanguageByTerseName("ecmascript"), Arrays.asList(ruleSet1));
47  
48          SidebarGenerator generator = new SidebarGenerator(writer, FileSystems.getDefault().getPath(".."));
49          List<Map<String, Object>> result = generator.generateRuleReferenceSection(rulesets);
50  
51          DumperOptions options = new DumperOptions();
52          options.setDefaultFlowStyle(FlowStyle.BLOCK);
53          if (SystemUtils.IS_OS_WINDOWS) {
54              options.setLineBreak(LineBreak.WIN);
55          }
56          String yaml = new Yaml(options).dump(result);
57  
58          String expected = MockedFileWriter.normalizeLineSeparators(
59                  IOUtils.toString(SidebarGeneratorTest.class.getResourceAsStream("sidebar.yml"), StandardCharsets.UTF_8));
60          assertEquals(expected, yaml);
61      }
62  }
