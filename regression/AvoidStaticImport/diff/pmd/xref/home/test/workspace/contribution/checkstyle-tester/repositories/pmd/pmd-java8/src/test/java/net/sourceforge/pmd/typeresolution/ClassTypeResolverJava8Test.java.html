45      @Test
46      public void testThisExpression() throws JaxenException {
47          ASTCompilationUnit acu = parseAndTypeResolveForClass18(ThisExpression.class);
48  
49          List<ASTPrimaryExpression> expressions = convertList(
50                  acu.findChildNodesWithXPath("//VariableInitializer/Expression/PrimaryExpression"),
51                  ASTPrimaryExpression.class);
52          List<ASTPrimaryPrefix> prefixes = convertList(
53                  acu.findChildNodesWithXPath("//VariableInitializer/Expression/PrimaryExpression/PrimaryPrefix"),
54                  ASTPrimaryPrefix.class);
55  
56          int index = 0;
57  
58          assertEquals(ThisExpression.class, expressions.get(index).getType());
59          assertEquals(ThisExpression.class, prefixes.get(index++).getType());
60          assertEquals(ThisExpression.PrimaryThisInterface.class, expressions.get(index).getType());
61          assertEquals(ThisExpression.PrimaryThisInterface.class, prefixes.get(index++).getType());
62  
63          // Make sure we got them all
64          assertEquals("All expressions not tested", index, expressions.size());
65          assertEquals("All expressions not tested", index, prefixes.size());
66      }
67  
68      @Test
69      public void testSuperExpression() throws JaxenException {
70          ASTCompilationUnit acu = parseAndTypeResolveForClass18(SuperExpression.class);
71  
72          List<AbstractJavaTypeNode> expressions = convertList(
73                  acu.findChildNodesWithXPath("//VariableInitializer/Expression/PrimaryExpression/PrimaryPrefix"),
74                  AbstractJavaTypeNode.class);
75  
76          int index = 0;
77  
78          assertEquals(SuperClass.class, expressions.get(index++).getType());
79  
80          // Make sure we got them all
81          assertEquals("All expressions not tested", index, expressions.size());
82      }
83  
84      private static <T> List<T> convertList(List<Node> nodes, Class<T> target) {
85          List<T> converted = new ArrayList<>();
86          for (Node n : nodes) {
87              converted.add(target.cast(n));
88          }
89          return converted;
90      }
91  
92      private ASTCompilationUnit parseAndTypeResolveForClass18(Class<?> clazz) {
93          return parseAndTypeResolveForClass(clazz, "1.8");
94      }
95  
96      // Note: If you're using Eclipse or some other IDE to run this test, you
97      // _must_ have the regress folder in
98      // the classpath. Normally the IDE doesn't put source directories themselves
99      // directly in the classpath, only
100     // the output directories are in the classpath.
101     private ASTCompilationUnit parseAndTypeResolveForClass(Class<?> clazz, String version) {
102         String sourceFile = clazz.getName().replace('.', '/') + ".java";
103         InputStream is = ClassTypeResolverJava8Test.class.getClassLoader().getResourceAsStream(sourceFile);
104         if (is == null) {
105             throw new IllegalArgumentException("Unable to find source file " + sourceFile + " for " + clazz);
106         }
107         LanguageVersionHandler languageVersionHandler = LanguageRegistry.getLanguage(JavaLanguageModule.NAME)
108                 .getVersion(version).getLanguageVersionHandler();
109         ASTCompilationUnit acu = (ASTCompilationUnit) languageVersionHandler
110                 .getParser(languageVersionHandler.getDefaultParserOptions()).parse(null, new InputStreamReader(is));
111         languageVersionHandler.getSymbolFacade().start(acu);
112         languageVersionHandler.getQualifiedNameResolutionFacade(ClassTypeResolverJava8Test.class.getClassLoader()).start(acu);
113         languageVersionHandler.getTypeResolutionFacade(ClassTypeResolverJava8Test.class.getClassLoader()).start(acu);
114         return acu;
115     }
