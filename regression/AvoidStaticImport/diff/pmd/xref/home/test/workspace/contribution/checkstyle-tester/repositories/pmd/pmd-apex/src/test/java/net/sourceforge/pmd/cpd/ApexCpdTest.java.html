7   import static org.junit.Assert.assertEquals;

9   
10  import java.io.File;
11  import java.io.IOException;
12  import java.util.Iterator;
13  
14  import org.apache.commons.io.FilenameUtils;
15  import org.junit.Before;
16  import org.junit.Test;
17  
18  import net.sourceforge.pmd.lang.apex.ApexLanguageModule;
19  
20  public class ApexCpdTest {
21      private File testdir;
22  
23      @Before
24      public void setUp() {
25          String path = FilenameUtils.normalize("src/test/resources/net/sourceforge/pmd/cpd/issue427");
26          testdir = new File(path);
27      }
28  
29      @Test
30      public void testIssue427() throws IOException {
31          CPDConfiguration configuration = new CPDConfiguration();
32          configuration.setMinimumTileSize(10);
33          configuration.setLanguage(LanguageFactory.createLanguage(ApexLanguageModule.TERSE_NAME));
34          CPD cpd = new CPD(configuration);
35          cpd.add(new File(testdir, "SFDCEncoder.cls"));
36          cpd.add(new File(testdir, "SFDCEncoderConstants.cls"));
37  
38          cpd.go();
39  
40          Iterator<Match> matches = cpd.getMatches();
41          int duplications = 0;
42          while (matches.hasNext()) {
43              matches.next();
44              duplications++;
45          }
46          assertEquals(1, duplications);
47          Match firstDuplication = cpd.getMatches().next();
48          assertTrue(firstDuplication.getSourceCodeSlice().startsWith("global with sharing class SFDCEncoder"));
49      }
50  }
