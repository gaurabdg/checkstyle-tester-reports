17  /**
18   *
19   * This package provides interfaces and classes dealing with step handling during propagation.
20   * It is used when a {@link org.orekit.propagation.Propagator Propagator} is run in
21   * {@link org.orekit.propagation.Propagator#MASTER_MODE master mode}. In this mode, the
22   * (master) propagator integration loop calls (slave) application callback methods at each
23   * finalized step. The callback methods must implement the {@link
24   * org.orekit.propagation.sampling.OrekitFixedStepHandler} interface for fixed step sampling or
25   * the {@link org.orekit.propagation.sampling.OrekitStepHandler} interface for variable step
26   * sampling.
27   *
28   * <p>
29   * The low level interfaces and classes (<code>OrekitXxx</code>)
30   * are heavily based on classes with similar names in the ode package from the <a
31   * href="https://hipparchus.org/">Hipparchus</a> library. The changes are mainly
32   * adaptations of the signatures to space dynamics:
33   * </p>
34   * <ul>
35   *  <li>the type of dependent variable t has been changed from <code>double</code>
36   *      to {@link org.orekit.time.AbsoluteDate AbsoluteDate}</li>
37   *  <li>the type of state vector y has been changed from <code>double[]</code>
38   *      to {@link org.orekit.propagation.SpacecraftState SpacecraftState}</li>
39   * </ul>
40   *
41   * @author Luc Maisonobe
42   *
43   */
44  package org.orekit.propagation.sampling;
