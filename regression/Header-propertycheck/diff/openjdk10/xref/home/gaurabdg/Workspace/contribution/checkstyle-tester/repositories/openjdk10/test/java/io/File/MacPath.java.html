24  /* @test
25   * @bug 7130915
26   * @summary Tests file path with nfc/nfd forms on MacOSX
27   * @requires (os.family == "mac")
28   * @library /test/lib
29   * @build jdk.test.lib.Asserts jdk.test.lib.process.ProcessTools MacPathTest
30   * @run main MacPath
31   */
32  
33  import java.util.Map;
34  
35  import jdk.test.lib.Asserts;
36  import jdk.test.lib.process.ProcessTools;
37  
38  public class MacPath {
39      public static void main(String args[]) throws Exception {
40          final ProcessBuilder pb =
41                  ProcessTools.createJavaProcessBuilder(true, MacPathTest.class.getName());
42          final Map<String, String> env = pb.environment();
43          env.put("LC_ALL", "en_US.UTF-8");
44          Process p = ProcessTools.startProcess("Mac Path Test", pb);
45          Asserts.assertTrue(p.waitFor() == 0, "test failed!");
46      }
47  }
