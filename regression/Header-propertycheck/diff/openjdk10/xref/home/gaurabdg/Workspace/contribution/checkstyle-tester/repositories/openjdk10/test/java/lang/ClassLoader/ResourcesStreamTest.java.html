24  import java.io.IOException;
25  import java.io.UncheckedIOException;
26  import java.net.URL;
27  import java.util.Collections;
28  import java.util.Enumeration;
29  import java.util.List;
30  import java.util.stream.Collectors;
31  import java.util.stream.Stream;
32  
33  /*
34   * @test
35   * @bug 8161230
36   * @summary Test java.lang.ClassLoader.resources() method
37   *
38   * @build ResourcesStreamTest
39   * @run main ResourcesStreamTest
40   */
41  public class ResourcesStreamTest {
42  
43      public static void main(String[] args) throws Exception {
44          testSuccess();
45          testFailure();
46      }
47  
48      public static void testSuccess() throws Exception {
49          // failing part first
50          try {
51              ClassLoader cl = new FailingClassLoader();
52              // should create the stream pipe
53              Stream<URL> stream = cl.resources("the name");
54              // expect function to throw an exception when calling the method
55              stream.forEach(System.out::println);
56              throw new Exception("expected UncheckedIOException not thrown");
57          } catch (UncheckedIOException uio) {
58              String causeMessage = uio.getCause().getMessage();
59              if (!"the name".equals(causeMessage))
60                  throw new Exception("unexpected cause message: " + causeMessage);
61          }
62      }
63  
64      public static void testFailure() throws Exception {
65          ClassLoader cl = new SuccessClassLoader();
66          long count = cl.resources("the name").count();
67          if (count != 1)
68              throw new Exception("expected resource is null or empty");
69  
70          cl.resources("the name")
71            .filter(url -> "file:/somefile".equals(url.toExternalForm()))
72            .findFirst()
73            .orElseThrow(() -> new Exception("correct URL not found"));
74      }
75  
76      public static class SuccessClassLoader extends ClassLoader {
77          @Override
78          public Enumeration<URL> getResources(String name) throws IOException {
79              URL url = new URL("file:/somefile");
80              return Collections.enumeration(Collections.singleton(url));
81          }
82      }
83  
84      public static class FailingClassLoader extends ClassLoader {
85          @Override
86          public Enumeration<URL> getResources(String name) throws IOException {
87              throw new IOException(name);
88          }
89      }
90  }
