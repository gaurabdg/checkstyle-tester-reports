24  import java.awt.*;
25  import java.awt.image.BufferedImage;
26  
27  
28  /*
29   * @test
30   * @summary Test the methods TrayIcon.getSize and SystemTray.getTrayIconSize.
31   *          There is no way to check whether the values returned are correct,
32   *          so its checked whether the value is greater than a minimum
33   * @author Dmitriy Ermashov (dmitriy.ermashov@oracle.com)
34   * @run main TrayIconSizeTest
35   */
36  
37  public class TrayIconSizeTest {
38  
39      public static void main(String[] args) throws Exception {
40          if (! SystemTray.isSupported()) {
41              System.out.println("SystemTray not supported on the platform under test. " +
42                      "Marking the test passed");
43          } else {
44              new TrayIconSizeTest().doTest();
45          }
46      }
47  
48      void doTest() throws Exception {
49  
50          SystemTray tray = SystemTray.getSystemTray();
51          Dimension dim = tray.getTrayIconSize();
52  
53          if (dim.width <= 5 || dim.height <= 5)
54              throw new RuntimeException("FAIL: value returned by getTrayIconSize is not correct: " + dim);
55  
56          TrayIcon icon = new TrayIcon(new BufferedImage(20, 20, BufferedImage.TYPE_INT_RGB));
57  
58          if (! icon.getSize().equals(dim))
59              throw new RuntimeException("FAIL: TrayIcon.getSize did not return the same value as " +
60                      "getTrayIconSize when TrayIcon not added" + "\n" +
61                      "SystemTray.getTrayIconSize(): " + dim + "\n" +
62                      "TrayIcon.getSize(): " + icon.getSize());
63  
64          tray.add(icon);
65  
66          if (icon.getSize().width <= 5 || icon.getSize().height <= 5)
67              throw new RuntimeException("FAIL: value returned by TrayIcon.getSize is not correct: " + icon.getSize());
68      }
69  }
