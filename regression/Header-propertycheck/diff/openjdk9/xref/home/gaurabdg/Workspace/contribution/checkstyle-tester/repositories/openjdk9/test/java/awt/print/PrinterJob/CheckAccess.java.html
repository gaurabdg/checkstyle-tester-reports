24  /**
25   * @test
26   * @bug 4151121
27   * @summary Confirm that PrinterJob.getPrinterJob is access checked.
28   * @author Graham Hamilton
29   */
30  
31  import java.awt.print.*;
32  import java.security.*;
33  
34  public class CheckAccess {
35  
36      static boolean verbose;
37  
38      private static void println(String mess) {
39          if (verbose) {
40              System.err.println(mess);
41          }
42      }
43  
44      /**
45       * SecurityManager that rejects all print requests,
46       * but allows everything else.
47       */
48      static class PrintHater extends SecurityManager {
49  
50          public void checkPermission(Permission p) {
51             // We're easy.
52          }
53  
54          public void checkPrintJobAccess() {
55              throw new SecurityException("No way!");
56          }
57      }
58  
59      public static void main(String argv[]) {
60  
61          if (argv.length > 0 && argv[0].equals("-v")) {
62              verbose = true;
63          }
64  
65          // Try to install our own security manager.
66          try {
67              SecurityManager sm = new PrintHater();
68              println("Installing PrintHater security manager");
69              System.setSecurityManager(sm);
70              println("Installed security manager OK");
71  
72          } catch (Throwable th) {
73              System.err.println("Failed to install SecurityManager");
74              th.printStackTrace();
75              throw new RuntimeException("Failed to install SecurityManager");
76          }
77  
78          try {
79  
80              println("Calling PrinterJob.getPrinterJob()");
81              PrinterJob.getPrinterJob();
82  
83              // Woops.  We did not get the SecurityException we expected.
84              println("Failed to get SecurityException");
85              throw new RuntimeException("Failed to get expected SecurityException");
86  
87          } catch (SecurityException ex) {
88              // Happy, happy.  This is what we want.
89              println("Got expected SecurityException OK.");
90              return;
91          }
92  
93      }
94  }
