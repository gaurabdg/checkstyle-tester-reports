26  /**
27   * Provides the classes necessary to create an applet and the classes an applet
28   * uses to communicate with its applet context.
29   * <p>
30   * The applet framework involves two entities: the <i>applet</i> and the
31   * <i>applet context</i>. An applet is an embeddable window (see the Panel
32   * class) with a few extra methods that the applet context can use to
33   * initialize, start, and stop the applet.
34   * <p>
35   * The applet context is an application that is responsible for loading and
36   * running applets. For example, the applet context could be a Web browser or an
37   * applet development environment.
38   * <p>
39   * The APIs in this package are all deprecated. Alternative technologies such as
40   * Java Web Start or installable applications should be used instead.
41   * See <a href="http://openjdk.java.net/jeps/289">JEP 289</a> and
42   * the Oracle White Paper
43   * <a href="http://www.oracle.com/technetwork/java/javase/migratingfromapplets-2872444.pdf">
44   * "Migrating from Java Applets to plugin-free Java technologies"</a> for more
45   * information.
46   *
47   * @since 1.0
48   */
49  package java.applet;
