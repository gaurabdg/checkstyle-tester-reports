23  /*
24   * @test
25   * @bug 7197401
26   * @summary Add a subset of the org.objectweb.asm packages to jdk8
27   * This test doesn't test asm functionality, it just tests the presence of
28   * asm in the jdk.
29   * These compile/run commands do the following:
30   * - Verify that asm is not in ct.sym so user code that refs it won't compile.
31   * - Verify that asm really is in rt.jar and can be accessed when ct.sym is not used.
32   * - Verify that if user code is compiled without ct.sym, it can't access asm classes
33   *   at runtime when a security manager is in use.
34   *
35   * @compile/fail AsmSanity.java
36   *
37   * @compile -XDignore.symbol.file=true AsmSanity.java
38   * @run main/othervm AsmSanity
39   *
40   * @run main/othervm/fail AsmSanity secure
41   *
42   */
43  
44  
45  // Verify that the expected asm pkgs are present
46  import jdk.internal.org.objectweb.asm.*;
47  import jdk.internal.org.objectweb.asm.commons.*;
48  import jdk.internal.org.objectweb.asm.signature.*;
49  import jdk.internal.org.objectweb.asm.tree.*;
50  import jdk.internal.org.objectweb.asm.tree.analysis.*;
51  import jdk.internal.org.objectweb.asm.util.*;
52  
53  // Verify that we can actually run some of the asm code.
54  public class AsmSanity {
55  
56      static public void main(String[] args) {
57          if (args.length == 0) {
58              System.out.println("-- Running without SecurityManager");
59              new Label();
60              System.out.println("-- Passed");
61              return;
62          }
63  
64          if (args[0].equals("secure")) {
65              System.out.println("-- Running with SecurityManager");
66              java.lang.SecurityManager sm = new SecurityManager();
67              System.setSecurityManager(sm);
68              // This should cause an accessClassInPackage exception
69              new Label();
70              return;
71          }
72          throw new Error("-- Failed:  Unknown argument to main: " + args[0]);
73      }
74  }
